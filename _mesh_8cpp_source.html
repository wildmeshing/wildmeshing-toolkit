25 std::vector<Tuple> Mesh::get_all(PrimitiveType type, const bool include_deleted) const
26 {
27  std::vector<Tuple> ret;
28 
29  if (static_cast<int8_t>(type) > top_cell_dimension()) return ret;
30 
31  const int64_t cap = capacity(type);
32 
33  const attribute::Accessor<char> flag_accessor = get_flag_accessor(type);
34  const attribute::CachingAccessor<char>& flag_accessor_indices = flag_accessor.index_access();
35  ret.reserve(cap);
36  for (size_t index = 0; index < cap; ++index) {
37  if (flag_accessor_indices.const_scalar_attribute(index) & 1)
38  ret.emplace_back(tuple_from_id(type, index));
39  else if (include_deleted)
40  ret.emplace_back();
41  }
42  return ret;
43 }
