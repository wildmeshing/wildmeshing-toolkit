123 const auto& v = simplex.vertices();
124
125 RawSimplexCollection sc;
126
127 if constexpr (N == 1) {
128 // do nothing
129 } else if constexpr (N == 2) {
130 sc.m_v.emplace_back(Vertex(v[0]));
131 sc.m_v.emplace_back(Vertex(v[1]));
132 } else if constexpr (N == 3) {
133 sc.m_v.emplace_back(Vertex(v[0]));
134 sc.m_v.emplace_back(Vertex(v[1]));

136 sc.m_e.emplace_back(Edge(v[0], v[1]));
137 sc.m_e.emplace_back(Edge(v[0], v[2]));
138 sc.m_e.emplace_back(Edge(v[1], v[2]));
139 } else {
140 static_assert(N == 4);
141
142 sc.m_v.emplace_back(Vertex(v[0]));
143 sc.m_v.emplace_back(Vertex(v[1]));
144 sc.m_v.emplace_back(Vertex(v[2]));
145 sc.m_v.emplace_back(Vertex(v[3]));
146 sc.m_e.emplace_back(Edge(v[0], v[1]));
147 sc.m_e.emplace_back(Edge(v[0], v[2]));
148 sc.m_e.emplace_back(Edge(v[0], v[3]));
149 sc.m_e.emplace_back(Edge(v[1], v[2]));
150 sc.m_e.emplace_back(Edge(v[1], v[3]));
151 sc.m_e.emplace_back(Edge(v[2], v[3]));
152 sc.m_f.emplace_back(Face(v[0], v[1], v[2]));
153 sc.m_f.emplace_back(Face(v[0], v[1], v[3]));
154 sc.m_f.emplace_back(Face(v[0], v[2], v[3]));
