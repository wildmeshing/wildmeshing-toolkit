172  template <class T, int order>
173  inline std::decay_t<T> quadrature_error_1pixel_eval(
174  const Eigen::Matrix<T, 2, 3> edge_verts,
175  std::function<T(const T&, const T&)> image_get_z,
176  wmtk::LineQuadrature& quad) const
177  {
178  quad.get_quadrature(order);
179  double ret = 0.0;
180  auto v1z = edge_verts(0, 2);
181  auto v2z = edge_verts(1, 2);
182  Eigen::Matrix<T, 1, 2> v12d, v22d;
183  v12d << edge_verts(0, 0), edge_verts(0, 1);
184  v22d << edge_verts(1, 0), edge_verts(1, 1);
185  // now do 1d quadrature
186  for (int i = 0; i < quad.points.rows(); i++) {
187  auto tmpu =
188  (1 - quad.points(i, 0)) * edge_verts(0, 0) + quad.points(i, 0) * edge_verts(1, 0);
189  auto tmpv =
190  (1 - quad.points(i, 0)) * edge_verts(0, 1) + quad.points(i, 0) * edge_verts(1, 1);
191  auto tmph = image_get_z(tmpu, tmpv);
192  auto tmpz = (1 - quad.points(i, 0)) * v1z + quad.points(i, 0) * v2z;
193  ret += abs(quad.weights(i) * (tmph - tmpz));
194  }
195 
196  return ret * (v12d - v22d).stableNorm();
197  }
198 };
