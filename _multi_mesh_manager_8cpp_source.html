893 void MultiMeshManager::update_vertex_operation_hashes_internal(
894  Mesh& m,
895  const Tuple& vertex,
896  Accessor<int64_t>& hash_accessor)
897 {
898  const PrimitiveType pt = m.top_simplex_type();
899  const simplex::SimplexCollection star =
900  simplex::closed_star(m, simplex::Simplex::vertex(vertex));
901  std::vector<Tuple> tuples_to_update;
902  switch (pt) {
903  case PrimitiveType::Vertex: {
904  const auto star_vertices = star.simplex_vector(PrimitiveType::Vertex);
905  tuples_to_update.reserve(star_vertices.size());
906  for (const simplex::Simplex& s : star_vertices) {
907  tuples_to_update.emplace_back(s.tuple());
908  }
909  break;
910  }
911  case PrimitiveType::Edge: {
912  const auto star_edges = star.simplex_vector(PrimitiveType::Edge);
913  tuples_to_update.reserve(star_edges.size());
914  for (const simplex::Simplex& s : star_edges) {
915  tuples_to_update.emplace_back(s.tuple());
916  }
917  break;
918  }
919  case PrimitiveType::Face: {
920  const auto star_faces = star.simplex_vector(PrimitiveType::Face);
921  tuples_to_update.reserve(star_faces.size());
922  for (const simplex::Simplex& s : star_faces) {
923  tuples_to_update.emplace_back(s.tuple());
924  }
925  break;
926  }
927  case PrimitiveType::Tetrahedron: {
928  const auto star_tets = star.simplex_vector(PrimitiveType::Tetrahedron);
929  tuples_to_update.reserve(star_tets.size());
930  for (const simplex::Simplex& s : star_tets) {
931  tuples_to_update.emplace_back(s.tuple());
932  }
933  break;
934  }
935  default:
936  case PrimitiveType::HalfEdge: throw std::runtime_error("Invalid case");
937  }
938 
939 
940  m.update_cell_hashes(tuples_to_update, hash_accessor);
941 
942  // need to get a star with new hash, otherwise cannot get_simplices()
943  auto vertex_new_hash = m.resurrect_tuple(vertex, hash_accessor);
944  const simplex::SimplexCollection star_new_hash =
945  simplex::closed_star(m, simplex::Simplex::vertex(vertex_new_hash));
946  update_vertex_operation_multimesh_map_hash_internal(m, star_new_hash, hash_accessor);
947 }
