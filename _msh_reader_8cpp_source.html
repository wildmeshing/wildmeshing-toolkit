283 void MshReader::extract_simplex_elements()
284 {
285  const auto* vertex_block = get_vertex_block(DIM);
286  const auto* element_block = get_simplex_element_block(DIM);
287  if (element_block == nullptr) return;
288  assert(vertex_block != nullptr);
289 
290  const size_t num_elements = element_block->num_elements_in_block;
291  if (num_elements == 0) return;
292  assert(vertex_block->num_nodes_in_block != 0);
293 
294  const size_t vert_tag_offset = vertex_block->tags.front();
295  const size_t elem_tag_offset = element_block->data.front();
296  for (size_t i = 0; i < num_elements; i++) {
297  const size_t tag = element_block->data[i * (DIM + 2)] - elem_tag_offset;
298  assert(tag < num_elements);
299  const auto* element = element_block->data.data() + i * (DIM + 2) + 1;
300 
301  if constexpr (DIM == 1) {
302  set_edge(tag, element[0] - vert_tag_offset, element[1] - vert_tag_offset);
303  } else if constexpr (DIM == 2) {
304  set_face(
305  tag,
306  element[0] - vert_tag_offset,
307  element[1] - vert_tag_offset,
308  element[2] - vert_tag_offset);
309  } else if constexpr (DIM == 3) {
310  set_tet(
311  tag,
312  element[0] - vert_tag_offset,
313  element[1] - vert_tag_offset,
314  element[2] - vert_tag_offset,

316  }

397 // for now, this function only supports reading a single attribute on the tet mesh
398 void MshReader::extract_element_attribute(
399  std::shared_ptr<wmtk::TetMesh> m,
400  const std::string& attr_name,
401  int DIM)
402 {
403  const auto* element_block = get_simplex_element_block(DIM);
404  const auto tuples = m->get_all(m->top_simplex_type());
405  const size_t tag_offset = element_block->data.front();
406 
407  for (const auto& data : m_spec.element_data) {
408  if (data.header.string_tags.front() != attr_name) {
409  continue;
410  }
411  assert(data.entries.size() == tuples.size());
412 
413  const int64_t attr_dim = data.header.int_tags[1];
414 
415  auto tag_handle =
416  m->register_attribute<double>(attr_name, PrimitiveType::Tetrahedron, attr_dim);
417  auto acc = m->create_accessor<double>(tag_handle);
418 
419  for (size_t i = 0; i < tuples.size(); ++i) {
420  const auto& entry = data.entries[i];
421  const size_t tag = entry.tag - tag_offset;
422  assert(tag < element_block->num_elements_in_block);
423  assert(tag == i);
424  if (attr_dim == 1) {
425  acc.scalar_attribute(tuples[i]) = entry.data[0];
426  } else {
427  acc.vector_attribute(tuples[i]) =
428  Eigen::Map<const Eigen::VectorXd>(entry.data.data(), entry.data.size());
429  }
430  }
431  return;
432  }
