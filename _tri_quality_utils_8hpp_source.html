19 // template get 3d tri area
20 template <typename T>
21 T triangle_3d_area(
22  const Eigen::Matrix<T, 3, 1>& a,
23  const Eigen::Matrix<T, 3, 1>& b,
24  const Eigen::Matrix<T, 3, 1>& c)
25 {
26  const T n0 = (a[1] - b[1]) * (a[2] - c[2]) - (a[1] - c[1]) * (a[2] - b[2]);
27  const T n1 = -(a[0] - b[0]) * (a[2] - c[2]) + (a[0] - c[0]) * (a[2] - b[2]);
28  const T n2 = (a[0] - b[0]) * (a[1] - c[1]) - (a[0] - c[0]) * (a[1] - b[1]);
29 
30  return sqrt(n0 * n0 + n1 * n1 + n2 * n2) * static_cast<T>(0.5);
31 }

34 template <typename T>
35 T triangle_2d_area(
36  const Eigen::Matrix<T, 2, 1>& A,
37  const Eigen::Matrix<T, 2, 1>& B,
38  const Eigen::Matrix<T, 2, 1>& C)
39 {
40  auto B_A = B - A;
41  auto C_A = C - A;
42  T area = static_cast<T>(0.5) * abs(B_A.x() * C_A.y() - B_A.y() * C_A.x());
43  return area;
44 }
