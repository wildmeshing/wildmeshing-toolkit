91  points_vector.push_back(std::pair<Vector3r, int64_t>(points[i], i));
92  }

94  // find the first colinear ABC with nonlinear BCD and delete C from vector
95  while (points_vector.size() > 3) {
96  bool no_colinear = true;
97  for (int64_t i = 0; i < points_vector.size(); ++i) {
98  auto cur = points_vector[i];
99  auto next = points_vector[(i + 1) % points_vector.size()];
100  auto prev = points_vector[(i + points_vector.size() - 1) % points_vector.size()];
101  auto nextnext = points_vector[(i + 2) % points_vector.size()];
102 
103  // TODO: check if orient2d == 0 works for check colinearity
104  // if (orient2d(prev.first.data(), cur.first.data(), next.first.data()) == 0 &&
105  // orient2d(cur.first.data(), next.first.data(), nextnext.first.data()) != 0) {
106 
107  Vector3r a = cur.first - prev.first;
108  Vector3r b = next.first - cur.first;
109  Vector3r c = nextnext.first - next.first;
110 
111  if (((a[0] * b[1] - a[1] * b[0]).get_sign() == 0 &&
112  (a[1] * b[2] - a[2] * b[1]).get_sign() == 0 &&
113  (a[0] * b[2] - a[2] * b[0]).get_sign() == 0) &&
114  (((b[0] * c[1] - b[1] * c[0]).get_sign() != 0 ||
115  (b[1] * c[2] - b[2] * c[1]).get_sign() != 0 ||
116  (b[0] * c[2] - b[2] * c[0]).get_sign() != 0))) {
117  no_colinear = false;
118  std::array<int64_t, 3> t = {{cur.second, next.second, nextnext.second}};
119  triangulated_faces.push_back(t);
120  points_vector.erase(points_vector.begin() + ((i + 1) % points_vector.size()));
121  break;
122  } else {
123  continue;
124  }
125  }
126 
127  if (no_colinear) break;
128  }
129 
130  // cleanup convex polygon
131  while (points_vector.size() >= 3) {
132  std::array<int64_t, 3> t = {
133  {points_vector[0].second,
134  points_vector[1].second,
135  points_vector[points_vector.size() - 1].second}};
136  triangulated_faces.push_back(t);
137  points_vector.erase(points_vector.begin());
138  }
139 
140  return triangulated_faces;
