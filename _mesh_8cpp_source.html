28 std::tuple<std::vector<std::vector<long>>, std::vector<std::vector<long>>> Mesh::consolidate()
29 {
30  // Number of dimensions
31  long tcp = top_cell_dimension() + 1;
32 
33  // Store the map from new indices to old. First index is dimensions, second simplex id
34  std::vector<std::vector<long>> new2old(tcp);
35  // Store the map from old indices to new. First index is dimensions, second simplex id
36  std::vector<std::vector<long>> old2new(tcp);
37 
38  // Initialize both maps
39  for (long d = 0; d < tcp; d++) {
40  Accessor<char> flag_accessor = get_flag_accessor(wmtk::get_primitive_type_from_id(d));
41  for (long i = 0; i < capacity(wmtk::get_primitive_type_from_id(d)); ++i) {
42  if (flag_accessor.index_access().scalar_attribute(i) & 1) {
43  old2new[d].push_back(new2old[d].size());
44  new2old[d].push_back(old2new[d].size() - 1); // -1 since we just pushed into it
45  } else {
46  old2new[d].push_back(-1);
47  }
48  }
49  }
50 
51  // Use new2oldmap to compact all attributes
52  for (long d = 0; d < tcp; d++) {
53  attribute::MeshAttributes<char>& attributesc = m_attribute_manager.m_char_attributes[d];
54  for (auto h = attributesc.m_attributes.begin(); h != attributesc.m_attributes.end(); h++)
55  h->consolidate(new2old[d]);
56 
57  attribute::MeshAttributes<long>& attributesl = m_attribute_manager.m_long_attributes[d];
58  for (auto h = attributesl.m_attributes.begin(); h != attributesl.m_attributes.end(); h++)
59  h->consolidate(new2old[d]);
60 
61  attribute::MeshAttributes<double>& attributesd = m_attribute_manager.m_double_attributes[d];
62  for (auto h = attributesd.m_attributes.begin(); h != attributesd.m_attributes.end(); h++)
63  h->consolidate(new2old[d]);
64 
65  attribute::MeshAttributes<Rational>& attributesr =
66  m_attribute_manager.m_rational_attributes[d];
67  for (auto h = attributesr.m_attributes.begin(); h != attributesr.m_attributes.end(); h++)
68  h->consolidate(new2old[d]);
69  }
70 
71  // Update the attribute size in the manager
72  for (long d = 0; d < tcp; d++) m_attribute_manager.m_capacities[d] = new2old[d].size();
73 
74  // Apply old2new to attributes containing indices
75  std::vector<std::vector<TypedAttributeHandle<long>>> handle_indices = connectivity_attributes();
76 
77  for (long d = 0; d < tcp; d++) {
78  for (long i = 0; i < handle_indices[d].size(); ++i) {
79  Accessor<long> accessor = create_accessor<long>(handle_indices[d][i]);
80  accessor.attribute().index_remap(old2new[d]);
81  }
82  }
83  // Return both maps for custom attribute remapping
84  return {new2old, old2new};
85 }
