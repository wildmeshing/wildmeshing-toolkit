56 // Add new data by giving the InputType
57 // InputType is used to make sure the pair of Input/Output is valid and to
58 // extract an id
59 // NOTE this passes value then key because the key can have variable arguments :(
60 template <typename InputType, typename ReturnType>
61 void add(ReturnType&& return_data, const InputType& input, const OtherArgumentTypes&... args)
62 {
63 using ReturnType_t = std::decay_t<ReturnType>;
64 if constexpr (std::is_same_v<ReturnType_t, void>) {
65 return;
66 }
67 static_assert(
68 !std::is_same_v<std::decay_t<InputType>, BaseType>,
69 "Don't pass in a input, use variant/visitor to get its "
70 "derived type");
71 // static_assert(
72 // !std::
73 // holds_alternative<std::reference_wrapper<std::decay_t<InputType>>,
74 // RefVariantType>,
75 // "the input type must be seen in the set of valid input variants");
76 // if the user passed in a input class lets try re-invoking with a
77 // derived type
78 auto id = get_id(input, args...);
79 using ExpectedReturnType = typename TypeHelper::template ReturnType<InputType>;
80
81 static_assert(
82 std::is_convertible_v<ReturnType_t, ExpectedReturnType>,
83 "Second argument should be the return value of a Functor "
84 "(or convertible at "
85 "least) ");
86
87 auto [it, did_insert] = m_data.try_emplace(
88 id,
89 ReturnVariant(
90 std::in_place_type_t<ExpectedReturnType>{},
91 std::forward<ReturnType>(return_data)));
92 if (!did_insert && m_enable_overwrites) {
93 throw std::runtime_error(
94 "Tried to overwite a value already stored in the return value cache");
95 }
96 }
