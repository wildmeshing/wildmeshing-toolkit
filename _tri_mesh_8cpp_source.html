96 Tuple TriMesh::switch_tuple(const Tuple& tuple, PrimitiveType type) const
97 {
98  assert(is_valid(tuple));
99  bool ccw = is_ccw(tuple);
100 
101  switch (type) {
102  case PrimitiveType::Triangle: {
103  const int64_t gvid = id(tuple, PrimitiveType::Vertex);
104  const int64_t geid = id(tuple, PrimitiveType::Edge);
105  const int64_t gfid = id(tuple, PrimitiveType::Triangle);
106 
107  auto ff = m_ff_accessor->const_vector_attribute<3>(tuple);
108 
109  int64_t gcid_new = ff(tuple.m_local_eid);
110  int64_t lvid_new = -1, leid_new = -1;
111 
112  auto fv = m_fv_accessor->index_access().const_vector_attribute<3>(gcid_new);
113 
114  auto fe = m_fe_accessor->index_access().const_vector_attribute<3>(gcid_new);
115 
116  if (gfid == gcid_new) {
117  // this supports 0,1,0 triangles not 0,0,0 triangles
118  int64_t oleid = tuple.m_local_eid;
119  int64_t olvid = tuple.m_local_vid;
120  for (int64_t i = 0; i < 3; ++i) {
121  if (i != oleid && fe(i) == geid) {
122  leid_new = i;
123  }
124  }
125  // if the old vertex is no "opposite of the old or new edges
126  // then they share the vertex
127  // a
128  // 0/ \1 <-- 0 and c share local ids, 1 and b share local ids
129  // /____\.
130  // b c
131  if (oleid != olvid && leid_new != olvid) {
132  lvid_new = olvid;
133  } else {
134  for (int64_t i = 0; i < 3; ++i) {
135  if (i != olvid && fv(i) == gvid) {
136  lvid_new = i;
137  }
138  }
139  }
140  } else {
141  for (int64_t i = 0; i < 3; ++i) {
142  if (fe(i) == geid) {
143  leid_new = i;
144  }
145  if (fv(i) == gvid) {
146  lvid_new = i;
147  }
148  }
149  }
150  assert(lvid_new != -1);
151  assert(leid_new != -1);
152 
153  const Tuple res(lvid_new, leid_new, tuple.m_local_fid, gcid_new);
154  assert(is_valid(res));
155  return res;
156  }
157  case PrimitiveType::Vertex:
158  case PrimitiveType::Edge: return autogen::tri_mesh::local_switch_tuple(tuple, type);

369 bool TriMesh::is_connectivity_valid() const
370 {
371  // get Accessors for topology
372  const attribute::Accessor<int64_t> fv_accessor = create_const_accessor<int64_t>(m_fv_handle);
373  const attribute::Accessor<int64_t> fe_accessor = create_const_accessor<int64_t>(m_fe_handle);
374  const attribute::Accessor<int64_t> ff_accessor = create_const_accessor<int64_t>(m_ff_handle);
375  const attribute::Accessor<int64_t> vf_accessor = create_const_accessor<int64_t>(m_vf_handle);
376  const attribute::Accessor<int64_t> ef_accessor = create_const_accessor<int64_t>(m_ef_handle);
377  const attribute::Accessor<char> v_flag_accessor = get_flag_accessor(PrimitiveType::Vertex);
378  const attribute::Accessor<char> e_flag_accessor = get_flag_accessor(PrimitiveType::Edge);
379  const attribute::Accessor<char> f_flag_accessor = get_flag_accessor(PrimitiveType::Triangle);
380 
381  // EF and FE
382  for (int64_t i = 0; i < capacity(PrimitiveType::Edge); ++i) {
383  if (e_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
384  wmtk::logger().debug("Edge {} is deleted", i);
385  continue;
386  }
387  int cnt = 0;
388  long ef_val = ef_accessor.index_access().const_scalar_attribute(i);
389 
390  auto fe_val = fe_accessor.index_access().const_vector_attribute<3>(ef_val);
391  for (int64_t j = 0; j < 3; ++j) {
392  if (fe_val(j) == i) {
393  cnt++;
394  }
395  }
396  if (cnt == 0) {
397  wmtk::logger().debug(
398  "EF[{0}] {1} and FE:[EF[{0}]] = {2} are not "
399  "compatible ",
400  i,
401  ef_val,
402  fmt::join(fe_val, ","));
403 
404  // std::cout << "EF and FE not compatible" << std::endl;
405  return false;
406  }
407  }
408 
409  // VF and FV
410  for (int64_t i = 0; i < capacity(PrimitiveType::Vertex); ++i) {
411  const int64_t vf = vf_accessor.index_access().const_scalar_attribute(i);
412  if (v_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
413  wmtk::logger().debug("Vertex {} is deleted", i);
414  continue;
415  }
416  int cnt = 0;
417 
418  auto fv = fv_accessor.index_access().const_vector_attribute<3>(vf);
419  for (int64_t j = 0; j < 3; ++j) {
420  if (fv(j) == i) {
421  cnt++;
422  }
423  }
424  if (cnt == 0) {
425  wmtk::logger().debug(
426  "VF and FV not compatible, could not find VF[{}] = {} "
427  "in FV[{}] = [{}]",
428  i,
429  vf,
430  vf,
431  fmt::join(fv, ","));
432  return false;
433  }
434  }
435 
436  // FE and EF
437  for (int64_t i = 0; i < capacity(PrimitiveType::Triangle); ++i) {
438  if (f_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
439  wmtk::logger().debug("Face {} is deleted", i);
440  continue;
441  }
442  auto fe = fe_accessor.index_access().const_vector_attribute<3>(i);
443  auto ff = ff_accessor.index_access().const_vector_attribute<3>(i);
444 
445  for (int64_t j = 0; j < 3; ++j) {
446  int neighbor_fid = ff(j);
447  const bool is_boundary = neighbor_fid == -1;
448  if (is_boundary) {
449  auto ef = ef_accessor.index_access().const_scalar_attribute(fe(j));
450  if (ef != i) {
451  wmtk::logger().debug(
452  "Even though local edge {} of face {} is "
453  "boundary (global eid is {}), "
454  "ef[{}] = {} != {}",
455  j,
456  i,
457  fe(j),
458  fe(j),
459  ef,
460  i);
461  return false;
462  }
463  } else {
464  if (neighbor_fid == i) {
465  logger().warn(
466  "Connectivity check cannot work when mapping a "
467  "face to itself (face {})",
468  i);
469  assert(false);
470  continue;
471  }
472  auto neighbor_ff =
473  ff_accessor.index_access().const_vector_attribute<3>(neighbor_fid);
474 
475  if ((neighbor_ff.array() == i).any()) {
476  auto neighbor_fe =
477  fe_accessor.index_access().const_vector_attribute<3>(neighbor_fid);
478 
479  int edge_shared_count = 0;
480  for (int local_neighbor_eid = 0; local_neighbor_eid < 3; ++local_neighbor_eid) {
481  // find some edge which is shared
482  if (neighbor_ff(local_neighbor_eid) == i) {
483  if (fe(j) == neighbor_fe(local_neighbor_eid)) {
484  edge_shared_count++;
485  }
486  }
487  }
488  if (edge_shared_count != 1) {
489  wmtk::logger().debug(
490  "face {} with fe={} neighbor fe[{}] = {} "
491  "was unable to find itself "
492  "uniquely (found {})",
493  i,
494  fmt::join(fe, ","),
495  neighbor_fid,
496  fmt::join(neighbor_fe, ","),
497  edge_shared_count);
498  return false;
499  }
500  } else {
501  wmtk::logger().debug(
502  "face {} with ff={} neighbor ff[{}] = {} was "
503  "unable to find itself",
504  i,
505  fmt::join(ff, ","),
506  neighbor_fid,
507  fmt::join(neighbor_ff, ","));
508  return false;
509  }
510  }
