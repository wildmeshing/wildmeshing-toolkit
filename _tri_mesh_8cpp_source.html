116 Tuple TriMesh::switch_tuple(const Tuple& tuple, PrimitiveType type) const
117 {

120 
121  switch (type) {
122  case PrimitiveType::Triangle: {
123  const int64_t gvid = id(tuple, PrimitiveType::Vertex);
124  const int64_t geid = id(tuple, PrimitiveType::Edge);
125  const int64_t gfid = id(tuple, PrimitiveType::Triangle);

133 
134  auto fe = m_fe_accessor->index_access().const_vector_attribute(gcid_new);
135 
136  if (gfid == gcid_new) {
137  // this supports 0,1,0 triangles not 0,0,0 triangles

139  int64_t olvid = tuple.m_local_vid;
140  for (int64_t i = 0; i < 3; ++i) {
141  if (i != oleid && fe(i) == geid) {
142  leid_new = i;
143  }
144  }
145  // if the old vertex is no "opposite of the old or new edges
146  // then they share the vertex
147  // a
148  // 0/ \1 <-- 0 and c share local ids, 1 and b share local ids
149  // /____\.
150  // b c
151  if (oleid != olvid && leid_new != olvid) {
152  lvid_new = olvid;
153  } else {
154  for (int64_t i = 0; i < 3; ++i) {

158  }
159  }

167  }
168  }
169  }

171  assert(leid_new != -1);

400 bool TriMesh::is_connectivity_valid() const
401 {
402  // get Accessors for topology
403  ConstAccessor<int64_t> fv_accessor = create_const_accessor<int64_t>(m_fv_handle);
404  ConstAccessor<int64_t> fe_accessor = create_const_accessor<int64_t>(m_fe_handle);
405  ConstAccessor<int64_t> ff_accessor = create_const_accessor<int64_t>(m_ff_handle);

408  ConstAccessor<char> v_flag_accessor = get_flag_accessor(PrimitiveType::Vertex);
409  ConstAccessor<char> e_flag_accessor = get_flag_accessor(PrimitiveType::Edge);
410  ConstAccessor<char> f_flag_accessor = get_flag_accessor(PrimitiveType::Triangle);
411 
412  // EF and FE
413  for (int64_t i = 0; i < capacity(PrimitiveType::Edge); ++i) {
414  if (e_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
415  wmtk::logger().debug("Edge {} is deleted", i);
416  continue;
417  }
418  int cnt = 0;
419  long ef_val = ef_accessor.index_access().const_scalar_attribute(i);
420 
421  auto fe_val = fe_accessor.index_access().const_vector_attribute(ef_val);
422  for (int64_t j = 0; j < 3; ++j) {
423  if (fe_val(j) == i) {
424  cnt++;
425  }
426  }
427  if (cnt == 0) {
428  wmtk::logger().debug(
429  "EF[{0}] {1} and FE:[EF[{0}]] = {2} are not "
430  "compatible ",
431  i,
432  ef_val,
433  fmt::join(fe_val, ","));
434 
435  // std::cout << "EF and FE not compatible" << std::endl;
436  return false;
437  }
438  }
439 
440  // VF and FV
441  for (int64_t i = 0; i < capacity(PrimitiveType::Vertex); ++i) {
442  const int64_t vf = vf_accessor.index_access().const_scalar_attribute(i);
443  if (v_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
444  wmtk::logger().debug("Vertex {} is deleted", i);
445  continue;
446  }
447  int cnt = 0;
448 
449  auto fv = fv_accessor.index_access().const_vector_attribute(vf);
450  for (int64_t j = 0; j < 3; ++j) {
451  if (fv(j) == i) {
452  cnt++;
453  }
454  }
455  if (cnt == 0) {
456  wmtk::logger().debug(
457  "VF and FV not compatible, could not find VF[{}] = {} "
458  "in FV[{}] = [{}]",
459  i,
460  vf,
461  vf,
462  fmt::join(fv, ","));
463  return false;
464  }
465  }
466 
467  // FE and EF
468  for (int64_t i = 0; i < capacity(PrimitiveType::Triangle); ++i) {
469  if (f_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
470  wmtk::logger().debug("Face {} is deleted", i);
471  continue;
472  }
473  auto fe = fe_accessor.index_access().const_vector_attribute(i);
474  auto ff = ff_accessor.index_access().const_vector_attribute(i);
475 
476  for (int64_t j = 0; j < 3; ++j) {
477  int neighbor_fid = ff(j);
478  const bool is_boundary = neighbor_fid == -1;
479  if (is_boundary) {
480  auto ef = ef_accessor.index_access().const_scalar_attribute(fe(j));
481  if (ef != i) {
482  wmtk::logger().debug(
483  "Even though local edge {} of face {} is "

485  "ef[{}] = {} != {}",
486  j,
487  i,
488  fe(j),
489  fe(j),
490  ef,
491  i);
492  return false;
493  }
494  } else {
495  if (neighbor_fid == i) {
496  logger().warn(
497  "Connectivity check cannot work when mapping a "
498  "face to itself (face {})",
499  i);
500  assert(false);

502  }
503  auto neighbor_ff = ff_accessor.index_access().const_vector_attribute(neighbor_fid);
504 
505  if ((neighbor_ff.array() == i).any()) {
506  auto neighbor_fe =
507  fe_accessor.index_access().const_vector_attribute(neighbor_fid);
508

511  // find some edge which is shared

513  if (fe(j) == neighbor_fe(local_neighbor_eid)) {
514  edge_shared_count++;
515  }
516  }
517  }
