747 #if defined(__cpp_concepts)
748 template <std::forward_iterator ContainerType>
749 #else
750 template <typename ContainerType>
751 #endif
752 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
753 {
754  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
755  Tuple r = tuple;
756  const PrimitiveType top_type = top_simplex_type();
757  for (const PrimitiveType primitive : sequence) {
758  // for top level simplices we cannot navigate across boundaries
759  if (primitive == top_type && is_boundary(r)) {
760  assert(!is_boundary(r));
761  r = {};
762  return r;
763  }
764  r = switch_tuple(r, primitive);
765  }
766  return r;
767 }
