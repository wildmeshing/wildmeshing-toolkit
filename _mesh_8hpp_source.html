861 #if defined(__cpp_concepts)
862 template <std::forward_iterator ContainerType>
863 #else
864 template <typename ContainerType>
865 #endif
866 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
867 {
868  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
869  Tuple r = tuple;
870  const PrimitiveType top_type = top_simplex_type();
871  for (const PrimitiveType primitive : sequence) {
872  // for top level simplices we cannot navigate across boundaries
873  if (primitive == top_type && is_boundary(r)) {
874  assert(!is_boundary(r));
875  r = {};
876  return r;
877  }
878  r = switch_tuple(r, primitive);
879  }
880  return r;
881 }
