287  auto run_single_queue = [&](auto& Q, int task_id) {
288  auto ele_in_queue = Elem();
289  while ([&]() { return Q.try_pop(ele_in_queue); }()) {
290  auto& [weight, op, tup, retry] = ele_in_queue;
291  if (!tup.is_valid(m)) continue;
292 
293  std::vector<Elem> renewed_elements;
294  {
295  auto locked_vid = lock_vertices(
296  m,
297  tup,
298  task_id); // Note that returning `Tuples` would be invalid.
299  if (!locked_vid) {
300  retry++;
301  if (retry < max_retry_limit) {
302  Q.emplace(ele_in_queue);
303  } else {
304  retry = 0;
305  final_queue.emplace(ele_in_queue);
306  }
307  continue;
308  }
309  if (tup.is_valid(m)) {
310  if (!is_weight_up_to_date(
311  m,
312  std::tuple<double, Op, Tuple>(
313  std::get<0>(ele_in_queue),
314  std::get<1>(ele_in_queue),
315  std::get<2>(ele_in_queue)))) {
316  operation_cleanup(m);
317  continue;
318  } // this can encode, in qslim, recompute(energy) == weight.
319  std::vector<std::pair<Op, Tuple>> renewed_tuples;
320  if constexpr (std::is_base_of<wmtk::TriMesh, AppMesh>::value) {
321  auto& my_op = (*new_edit_operation_maps[op]);
322  const bool success = my_op(m, tup);
323  if (success) {
324  // TODO: renew_neighbor_tuples should allow for triangles, edges, or
325  // vertices
326  renewed_tuples =
327  renew_neighbor_tuples(m, op, my_op.modified_triangles(m));
328  cnt_success++;
329  cnt_update++;
330  } else {
331  on_fail(m, op, tup);
332  cnt_fail++;
333  }
334  } else {
335  auto newtup = edit_operation_maps[op](m, tup);
336  if (newtup) {
337  renewed_tuples = renew_neighbor_tuples(m, op, newtup.value());
338  cnt_success++;
339  cnt_update++;
340  } else {
341  on_fail(m, op, tup);
342  cnt_fail++;
343  }
344  }
345 
346  for (auto& [o, e] : renewed_tuples) {
347  auto val = priority(m, o, e);
348  if (should_renew(val)) renewed_elements.emplace_back(val, o, e, 0);
349  }
350  }
351  operation_cleanup(m); // Maybe use RAII
352  }
353  for (auto& e : renewed_elements) {
354  Q.emplace(e);
355  }
356 
357  if (stop.load(std::memory_order_acquire)) return;
358  if (cnt_success > stopping_criterion_checking_frequency) {
359  if (stopping_criterion(m)) {
360  stop.store(true);
361  return;
362  }
363  cnt_update.store(0, std::memory_order_release);
364  }
365  FrameMark;
366  }
367  };
368 
369  if constexpr (policy == ExecutionPolicy::kSeq) {
370  for (auto& [op, e] : operation_tuples) {
371  if (!e.is_valid(m)) continue;
372  final_queue.emplace(priority(m, op, e), op, e, 0);
373  }
374  run_single_queue(final_queue, 0);
375  } else {
376  for (auto& [op, e] : operation_tuples) {
377  if (!e.is_valid(m)) continue;
378  queues[get_partition_id(m, e)].emplace(priority(m, op, e), op, e, 0);
379  }
380  // Comment out parallel: work on serial first.
381  tbb::task_arena arena(num_threads);
382  tbb::task_group tg;
383  arena.execute([&queues, &run_single_queue, &tg]() {
384  for (int task_id = 0; task_id < queues.size(); task_id++) {
385  tg.run([&run_single_queue, &queues, task_id] {
386  run_single_queue(queues[task_id], task_id);
387  });
388  }
389  tg.wait();
390  });
391  logger().debug("Parallel Complete, remains element {}", final_queue.size());
392  run_single_queue(final_queue, 0);
393  }
