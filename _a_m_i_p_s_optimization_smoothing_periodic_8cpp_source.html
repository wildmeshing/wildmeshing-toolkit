102template <int S>
103void AMIPSOptimizationSmoothingPeriodic::WMTKAMIPSProblem<S>::gradient(
104 const TVector& x,
105 TVector& gradv)
106{
107 constexpr int64_t size = S == 6 ? 2 : 3;
108 gradv.resize(size);
109 gradv.setZero();
110 Eigen::Matrix<double, size, 1> tmp(size);
111#ifdef WMTK_ENABLE_P
112 double tmp2 = 0;
113#endif
114
115 for (auto c : m_cells) {
116 if constexpr (S == 6) {
117 assert(x.size() == 2);
118 c[0] = x[0];
119 c[1] = x[1];
120 wmtk::function::Tri_AMIPS_jacobian(c, tmp);
121#ifdef WMTK_ENABLE_P
122 tmp2 = wmtk::function::Tri_AMIPS_energy(c);
123#endif
124 } else {
125 assert(x.size() == 3);
126 c[0] = x[0];
127 c[1] = x[1];
128 c[2] = x[2];
129 wmtk::function::Tet_AMIPS_jacobian(c, tmp);
130#ifdef WMTK_ENABLE_P
131 tmp2 = wmtk::function::Tet_AMIPS_energy(c);
132#endif
133 }
134#ifdef WMTK_ENABLE_P
135 gradv += double(p) * pow(tmp2, p - 1) * tmp;
136#else
137 gradv += tmp;
138#endif
139 }

142template <int S>
143void AMIPSOptimizationSmoothingPeriodic::WMTKAMIPSProblem<S>::hessian(
144 const TVector& x,
145 Eigen::MatrixXd& hessian)
146{
147 // std::cout << "error here !!! Hessian used" << std::endl;
148 constexpr int64_t size = S == 6 ? 2 : 3;
149 hessian.resize(size, size);
150 hessian.setZero();
151 Eigen::Matrix<double, size, size> tmp;
152#ifdef WMTK_ENABLE_P
153 double tmp2 = 0;
154 Eigen::Matrix<double, size, 1> tmpj(size);
155#endif
156
157 for (auto c : m_cells) {
158 if constexpr (S == 6) {
159 assert(x.size() == 2);
160 c[0] = x[0];
161 c[1] = x[1];
162 wmtk::function::Tri_AMIPS_hessian(c, tmp);
163#ifdef WMTK_ENABLE_P
164 wmtk::function::Tri_AMIPS_jacobian(c, tmpj);
165 tmp2 = wmtk::function::Tri_AMIPS_energy(c);
166#endif
167 } else {
168 assert(x.size() == 3);
169 c[0] = x[0];
170 c[1] = x[1];
171 c[2] = x[2];
172 wmtk::function::Tet_AMIPS_hessian(c, tmp);
173#ifdef WMTK_ENABLE_P
174 wmtk::function::Tet_AMIPS_jacobian(c, tmpj);
175 tmp2 = wmtk::function::Tet_AMIPS_energy(c);
176#endif
177 }
178#ifdef WMTK_ENABLE_P
179 hessian += (p) * (p - 1) * std::pow(tmp2, p - 2) * tmpj * tmpj.transpose() +
180 p * std::pow(tmp2, p - 1) * tmp;
181#else
182 hessian += tmp;
183#endif
184 // 12 f(x)^2 * f(x)'*f^T'(x) + 4 f(x)^3 * H
185 }
