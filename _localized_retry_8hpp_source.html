37template <class Mesh>
38size_t run_localized_to_convergence(
39 Mesh& m,
40 ExecutePass<Mesh>& executor,
41 std::vector<std::pair<Op, typename Mesh::Tuple>> ops)
42{
43 using Tuple = typename Mesh::Tuple;
44
45 // vertex_epoch[v] = the last round in which v was in some successful operation's
46 // modified region. Capacity is fixed within a phase (storage is preallocated), and
47 // any vertex created during the phase (splits) has an id below this capacity.
48 std::vector<uint64_t> vertex_epoch(m.vert_capacity(), 0);
49 uint64_t round = 0;
50 threading::collector<std::pair<Op, Tuple>> failures;
51
52 auto edge_epoch = [&vertex_epoch](const Mesh& m_, const Tuple& t) -> uint64_t {
53 const size_t a = t.vid(m_);
54 const size_t b = t.switch_vertex(m_).vid(m_);
55 const uint64_t ea = a < vertex_epoch.size() ? vertex_epoch[a] : 0;
56 const uint64_t eb = b < vertex_epoch.size() ? vertex_epoch[b] : 0;
57 return std::max(ea, eb);
58 };
59
60 // Wrap the driver-provided renewal: keep its behavior (re-enqueue affected tuples
61 // within the pass) and additionally stamp those tuples' vertices with the current
62 // round so the between-pass filter can find the failures adjacent to them.
63 auto driver_renew = executor.renew_neighbor_tuples;
64 executor.renew_neighbor_tuples =
65 [&, driver_renew](const Mesh& m_, Op op, const std::vector<Tuple>& newts) {
66 auto tups = driver_renew(m_, op, newts);
67 for (const auto& [_, t] : tups) {
68 const size_t a = t.vid(m_);
69 const size_t b = t.switch_vertex(m_).vid(m_);
70 if (a < vertex_epoch.size()) {
71 // this is thread-safe because each vertex is only ever modified by one
72 // operation at a time (the two-ring lock)
73 vertex_epoch[a] = round;
74 }
75 if (b < vertex_epoch.size()) {
76 vertex_epoch[b] = round;
77 }
78 }
79 return tups;
80 };
81 executor.on_fail = [&failures](const Mesh&, Op op, const Tuple& t) {
82 failures.emplace_back(op, t);
83 };
84
85 size_t total_success = 0;
86 do {
87 ++round;
88 failures.clear();
89 executor(m, ops);
90 total_success += static_cast<size_t>(executor.get_cnt_success());
91 ops.clear();
92 for (const auto& pr : failures) {
93 const Tuple& t = pr.second;
94 if (!t.is_valid(m)) {
95 continue;
96 }
97 // retry only if this failure's neighborhood was modified during this round
98 if (edge_epoch(m, t) == round) {
99 ops.emplace_back(pr);
100 }
101 }
102 } while (executor.get_cnt_success() > 0 && !ops.empty());
103 return total_success;
104}
