101  auto todo_attribute = m_mesh.register_attribute_typed<int64_t>(
102  "todo_edgesplit_in_marching_component",
103  wmtk::PrimitiveType::Edge,
104  1);
105  auto splitted_edges_attribute = m_mesh.register_attribute_typed<int64_t>(
106  "splitted_edges_in_marching_component",
107  wmtk::PrimitiveType::Edge,
108  1);
109 
110  assert(m_input_values.size() == 2);
111  const int64_t vertex_tag_0 = m_input_values[0];
112  const int64_t vertex_tag_1 = m_input_values[1];
113  wmtk::attribute::Accessor<int64_t> acc_vertex_tag =
114  m_mesh.create_accessor<int64_t>(m_vertex_tag_handle);
115 
116  std::deque<TagAttribute> filters;
117  for (size_t i = 0; i < m_filter_labels.size(); ++i) {
118  filters.emplace_back(m_mesh, m_filter_labels[i], PrimitiveType::Edge, m_filter_values[i]);
119  }
120 
121  // compute the todo list for the split edge
122  wmtk::attribute::Accessor<int64_t> acc_todo = m_mesh.create_accessor(todo_attribute);
123  wmtk::attribute::Accessor<int64_t> acc_splitted_edges =
124  m_mesh.create_accessor(splitted_edges_attribute);
125  for (const Tuple& edge : m_mesh.get_all(PrimitiveType::Edge)) {
126  bool is_of_interest = true;
127  for (const TagAttribute& filter : filters) {
128  if (filter.m_accessor.const_scalar_attribute(edge) != filter.m_val) {
129  is_of_interest = false;
130  break;
131  }
132  }
133 
134  if (!is_of_interest) {
135  continue;
136  }
137 
138  const int64_t vt0 = acc_vertex_tag.scalar_attribute(edge);
139  const int64_t vt1 =
140  acc_vertex_tag.scalar_attribute(m_mesh.switch_tuple(edge, PrimitiveType::Vertex));
141  if ((vt0 == vertex_tag_0 && vt1 == vertex_tag_1) ||
142  (vt1 == vertex_tag_0 && vt0 == vertex_tag_1)) {
143  acc_todo.scalar_attribute(edge) = 1;
144  acc_splitted_edges.scalar_attribute(edge) = 1;
145  }
146  }
147 
148  EdgeSplit op_split(m_mesh);
149  op_split.add_invariant(std::make_shared<TodoInvariant>(m_mesh, todo_attribute));
150 
151  // labels
152  {
153  /**************************edge tag******************************/
154  if (m_edge_tag_handle.has_value()) {
155  auto compute_edge_label =
156  [this](const Eigen::MatrixX<int64_t>& labels) -> Eigen::VectorX<int64_t> {
157  assert(labels.cols() == 2);
158  if (labels(0, 0) == m_output_value && labels(0, 1) == m_output_value)
159  return Eigen::VectorX<int64_t>::Constant(1, m_output_value);
160  return Eigen::VectorX<int64_t>::Constant(1, 0);
161  };
162 
163  std::shared_ptr edge_tag_strategy = std::make_shared<
164  wmtk::operations::SingleAttributeTransferStrategy<int64_t, int64_t>>(
165  m_edge_tag_handle.value(),
166  m_vertex_tag_handle,
167  compute_edge_label);
168 
169  op_split.add_transfer_strategy(edge_tag_strategy);
170  op_split.set_new_attribute_strategy(m_edge_tag_handle.value());
171  }
172 
173  /**************************face tag******************************/
174  if (m_face_tag_handle.has_value() && m_edge_tag_handle.has_value()) {
175  auto compute_face_label =
176  [this](const Eigen::MatrixX<int64_t>& labels) -> Eigen::VectorX<int64_t> {
177  assert(labels.cols() == 3);
178  if (labels(0, 0) == m_output_value && labels(0, 1) == m_output_value &&
179  labels(0, 2) == m_output_value) {
180  return Eigen::VectorX<int64_t>::Constant(1, m_output_value);
181  }
182  return Eigen::VectorX<int64_t>::Constant(1, 0);
183  };
184 
185  std::shared_ptr face_tag_strategy = std::make_shared<
186  wmtk::operations::SingleAttributeTransferStrategy<int64_t, int64_t>>(
187  m_face_tag_handle.value(),
188  m_edge_tag_handle.value(),
189  compute_face_label);
190 
191  op_split.add_transfer_strategy(face_tag_strategy);
192  op_split.set_new_attribute_strategy(m_face_tag_handle.value());
193  }
194 
195 
196  auto tmp = std::make_shared<SplitNewAttributeStrategy<int64_t>>(m_vertex_tag_handle);
197  tmp->set_strategy(SplitBasicStrategy::None);
198  tmp->set_rib_strategy(
199  [this](const VectorX<int64_t>&, const VectorX<int64_t>&, const std::bitset<2>&) {
200  VectorX<int64_t> ret(1);
201  ret(0) = m_output_value;
202  return ret;
203  });
204  op_split.set_new_attribute_strategy(m_vertex_tag_handle, tmp);
205 
206  op_split.set_new_attribute_strategy(
207  attribute::MeshAttributeHandle(m_mesh, splitted_edges_attribute),
208  SplitBasicStrategy::Copy,
209  SplitRibBasicStrategy::None);
210  }
211 
212  // filters
213  for (const auto& edge_filter_handle : m_filter_labels) {
214  op_split.set_new_attribute_strategy(
215  edge_filter_handle,
216  SplitBasicStrategy::None,
217  SplitRibBasicStrategy::None);
218  }
219  // todo_attribute
220  op_split.set_new_attribute_strategy(
221  attribute::MeshAttributeHandle(m_mesh, todo_attribute),
222  SplitBasicStrategy::None,
223  SplitRibBasicStrategy::None);
224  // pass_through
225  for (const auto& attr : m_pass_through_attributes) {
226  op_split.set_new_attribute_strategy(attr);
227  }
228 
229  Scheduler scheduler;
230  while (true) {
231  const auto stats = scheduler.run_operation_on_all(op_split);
232  if (stats.number_of_successful_operations() == 0) {
233  break;
234  }
235  }
236 
237  // move vertices according to weight
238  auto pos_handle = m_mesh.get_attribute_handle<double>("vertices", PrimitiveType::Vertex);
239  auto pos_acc = m_mesh.create_accessor<double>(pos_handle);
240  for (const Tuple& v : m_mesh.get_all(PrimitiveType::Vertex)) {
241  if (acc_vertex_tag.const_scalar_attribute(v) != m_output_value) {
242  continue;
243  }
244  // get the two input positions
245  Tuple v0;
246  Tuple v1;
247  auto incident_edges = simplex::cofaces_single_dimension_tuples(
248  m_mesh,
249  simplex::Simplex::vertex(m_mesh, v),
250  PrimitiveType::Edge);
251  for (Tuple e : incident_edges) {
252  if (acc_splitted_edges.const_scalar_attribute(e) == 0) {
253  continue;
254  }
255  if (acc_vertex_tag.const_scalar_attribute(e) == m_output_value) {
256  e = m_mesh.switch_tuple(e, PrimitiveType::Vertex);
257  }
258  if (acc_vertex_tag.const_scalar_attribute(e) == vertex_tag_0) {
259  v0 = e;
260  }
261  if (acc_vertex_tag.const_scalar_attribute(e) == vertex_tag_1) {
262  v1 = e;
263  }
264  }
265 #if defined(WMTK_ENABLE_HASH_UPDATE)
266  assert(m_mesh.is_valid_with_hash(v0));
267  assert(m_mesh.is_valid_with_hash(v1));
268 #else
269  assert(m_mesh.is_valid(v0));
270  assert(m_mesh.is_valid(v1));
271 #endif
272  const auto p0 = pos_acc.const_vector_attribute(v0);
273  const auto p1 = pos_acc.const_vector_attribute(v1);
274  pos_acc.vector_attribute(v) = (1 - m_weight) * p0 + m_weight * p1;
275  }
