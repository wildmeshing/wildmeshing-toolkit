27 std::tuple<std::vector<std::vector<int64_t>>, std::vector<std::vector<int64_t>>> Mesh::consolidate()
28 {
29  // Number of dimensions
30  int64_t tcp = top_cell_dimension() + 1;
31 
32  // Store the map from new indices to old. First index is dimensions, second simplex id
33  std::vector<std::vector<int64_t>> new2old(tcp);
34  // Store the map from old indices to new. First index is dimensions, second simplex id
35  std::vector<std::vector<int64_t>> old2new(tcp);
36 
37  // Initialize both maps
38  for (int64_t d = 0; d < tcp; d++) {
39  Accessor<char> flag_accessor = get_flag_accessor(wmtk::get_primitive_type_from_id(d));
40  for (int64_t i = 0; i < capacity(wmtk::get_primitive_type_from_id(d)); ++i) {
41  if (flag_accessor.index_access().scalar_attribute(i) & 1) {
42  old2new[d].push_back(new2old[d].size());
43  new2old[d].push_back(old2new[d].size() - 1); // -1 since we just pushed into it
44  } else {
45  old2new[d].push_back(-1);
46  }
47  }
48  }
49 
50  // Use new2oldmap to compact all attributes
51  for (int64_t d = 0; d < tcp; d++) {
52  attribute::MeshAttributes<char>& attributesc = m_attribute_manager.m_char_attributes[d];
53  for (auto h = attributesc.m_attributes.begin(); h != attributesc.m_attributes.end(); h++)
54  h->consolidate(new2old[d]);
55 
56  attribute::MeshAttributes<int64_t>& attributesl = m_attribute_manager.m_long_attributes[d];
57  for (auto h = attributesl.m_attributes.begin(); h != attributesl.m_attributes.end(); h++)
58  h->consolidate(new2old[d]);
59 
60  attribute::MeshAttributes<double>& attributesd = m_attribute_manager.m_double_attributes[d];
61  for (auto h = attributesd.m_attributes.begin(); h != attributesd.m_attributes.end(); h++)
62  h->consolidate(new2old[d]);
63 
64  attribute::MeshAttributes<Rational>& attributesr =
65  m_attribute_manager.m_rational_attributes[d];
66  for (auto h = attributesr.m_attributes.begin(); h != attributesr.m_attributes.end(); h++)
67  h->consolidate(new2old[d]);
68  }
69 
70  // Update the attribute size in the manager
71  for (int64_t d = 0; d < tcp; d++) m_attribute_manager.m_capacities[d] = new2old[d].size();
72 
73  // Apply old2new to attributes containing indices
74  std::vector<std::vector<TypedAttributeHandle<int64_t>>> handle_indices =
75  connectivity_attributes();
76 
77  for (int64_t d = 0; d < tcp; d++) {
78  for (int64_t i = 0; i < handle_indices[d].size(); ++i) {
79  Accessor<int64_t> accessor = create_accessor<int64_t>(handle_indices[d][i]);
80  accessor.attribute().index_remap(old2new[d]);
81  }
82  }
83  // Return both maps for custom attribute remapping
84  return {new2old, old2new};
85 }
