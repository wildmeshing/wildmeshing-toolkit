21  /\\ /\\
22  /ot\\ --> / \\ --> ...
23  /____\\ /____\\
24  */

32

34  if (m_mesh.id_tet(iter_tuple) == m_mesh.id_tet(t)) {

37  }
38

40  iter_tuple = m_mesh.switch_face(iter_tuple);
41  incident_tets.emplace_back(iter_tuple);

47  // go to an one boundary and start there
48  // direction
49  /*
50  /\\ /\\
51  / \\ --> /ot\\ --> ...
52  /____\\ /____\\
53  */

55

57  incident_faces.clear();
