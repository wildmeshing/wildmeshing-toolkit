258  template <class ACTION>
259  void bbox_intersect_recursive(
260  ACTION& action,
261  const Box& box,
262  index_t node, index_t b, index_t e
263  ) const {
264  geo_debug_assert(e != b);
265 
266  // Prune sub-tree that does not have intersection
267  if(!bboxes_overlap(box, bboxes_[node])) {
268  return;
269  }
270 
271  // Leaf case
272  if(e == b+1) {
273  action(b);
274  return;
275  }
276 
277  // Recursion
278  index_t m = b + (e - b) / 2;
279  index_t node_l = 2 * node;
280  index_t node_r = 2 * node + 1;
281 
282  bbox_intersect_recursive(action, box, node_l, b, m);
283  bbox_intersect_recursive(action, box, node_r, m, e);
284  }

307  template <class ACTION>
308  void intersect_recursive(
309  ACTION& action,
310  index_t node1, index_t b1, index_t e1,
311  index_t node2, index_t b2, index_t e2
312  ) const {
313  geo_debug_assert(e1 != b1);
314  geo_debug_assert(e2 != b2);
315 
316  // Since we are intersecting the AABBTree with *itself*,
317  // we can prune half of the cases by skipping the test
318  // whenever node2's facet index interval is greated than
319  // node1's facet index interval.
320  if(e2 <= b1) {
321  return;
322  }
323 
324  // The acceleration is here:
325  if(!bboxes_overlap(bboxes_[node1], bboxes_[node2])) {
326  return;
327  }
328 
329  // Simple case: leaf - leaf intersection.
330  if(b1 + 1 == e1 && b2 + 1 == e2) {
331  action(b1, b2);
332  return;
333  }
334 
335  // If node2 has more facets than node1, then
336  // intersect node2's two children with node1
337  // else
338  // intersect node1's two children with node2
339  if(e2 - b2 > e1 - b1) {
340  index_t m2 = b2 + (e2 - b2) / 2;
341  index_t node2_l = 2 * node2;
342  index_t node2_r = 2 * node2 + 1;
343  intersect_recursive(action, node1, b1, e1, node2_l, b2, m2);
344  intersect_recursive(action, node1, b1, e1, node2_r, m2, e2);
345  } else {
346  index_t m1 = b1 + (e1 - b1) / 2;
347  index_t node1_l = 2 * node1;
348  index_t node1_r = 2 * node1 + 1;
349  intersect_recursive(action, node1_l, b1, m1, node2, b2, e2);
350  intersect_recursive(action, node1_r, m1, e1, node2, b2, e2);
351  }
352  }
