673 #if defined(__cpp_concepts)
674 template <std::forward_iterator ContainerType>
675 #else
676 template <typename ContainerType>
677 #endif
678 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
679 {
680  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
681  Tuple r = tuple;
682  const PrimitiveType top_type = top_simplex_type();
683  for (const PrimitiveType primitive : sequence) {
684  // for top level simplices we cannot navigate across boundaries
685  if (primitive == top_type && is_boundary(r)) {
686  assert(!is_boundary(r));
687  r = {};
688  return r;
689  }
690  r = switch_tuple(r, primitive);
691  }
692  return r;
693 }
