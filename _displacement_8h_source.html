75  template <class T>
76  inline T triangle_3d_area(
77  const Eigen::Matrix<T, 3, 1>& a,
78  const Eigen::Matrix<T, 3, 1>& b,
79  const Eigen::Matrix<T, 3, 1>& c)
80  {
81  const T n0 = (a(1) - b(1)) * (a(2) - c(2)) - (a(1) - c(1)) * (a(2) - b(2));
82  const T n1 = -(a(0) - b(0)) * (a(2) - c(2)) + (a(0) - c(0)) * (a(2) - b(2));
83  const T n2 = (a(0) - b(0)) * (a(1) - c(1)) - (a(0) - c(0)) * (a(1) - b(1));
84 
85  return sqrt(n0 * n0 + n1 * n1 + n2 * n2) * static_cast<T>(0.5);
86  };

114  template <class T, int order>
115  inline T get_error_per_edge_T(
116  const Eigen::Matrix<T, 2, 1>& uv1,
117  const Eigen::Matrix<T, 2, 1>& uv2) const
118  {
119  throw std::runtime_error("get_error_per_edge should not be called");
120  auto p1_displaced = get(uv1(0), uv1(1));
121  auto p2_displaced = get(uv2(0), uv2(1));
122  // get the pixel index of p1 and p2
123  auto [xx1, yy1] = get_coordinate(get_value(uv1(0)), get_value(uv1(1)));
124  auto [xx2, yy2] = get_coordinate(get_value(uv2(0)), get_value(uv2(1)));
125  // get all the pixels in between p1 and p2
126  auto pixel_num = std::max(abs(xx2 - xx1), abs(yy2 - yy1));
127  if (pixel_num <= 0) return T(0.);
128  assert(pixel_num > 0);
129  T error = T(0.0);
130  auto lerp = [](const auto& a, const auto& b, const T& t) { return (1. - t) * a + t * b; };
131  for (int i = 0; i < pixel_num; i++) {
132  const auto r0 = T(i) / pixel_num;
133  const auto r1 = T(i + 1) / pixel_num;
134  Eigen::Matrix<T, 2, 1> pixel_uv1 = lerp(uv1, uv2, r0);
135  Eigen::Matrix<T, 2, 1> pixel_uv2 = lerp(uv1, uv2, r1);
136  Eigen::Matrix<T, 3, 1> pixel_p1 = lerp(p1_displaced, p2_displaced, r0);
137  Eigen::Matrix<T, 3, 1> pixel_p2 = lerp(p1_displaced, p2_displaced, r1);
138  wmtk::LineQuadrature quad;
139  quad.get_quadrature(order);
140 
141  T unweighted_error = T(0.0);
142  for (int j = 0; j < quad.points.rows(); j++) {
143  const T t = T(quad.points(j, 0));
144  Eigen::Matrix<T, 2, 1> tmpuv = lerp(pixel_uv1, pixel_uv2, t);
145  Eigen::Matrix<T, 3, 1> tmpp_displaced = get(tmpuv(0), tmpuv(1));
146 
147  Eigen::Matrix<T, 3, 1> tmpp_tri = lerp(pixel_p1, pixel_p2, t);
148  Eigen::Matrix<T, 3, 1> diffp = tmpp_tri - tmpp_displaced;
149  unweighted_error += T(quad.weights(j)) * diffp.norm();
150  }
151  auto diffuv = pixel_uv1 - pixel_uv2;
152  error += unweighted_error * diffuv.norm();
153  }
154  assert(error >= 0);
155  return error;
156  }

177  template <class T>
178  inline T get_error_per_triangle_T(const Eigen::Matrix<T, 3, 2, Eigen::RowMajor>& triangle)
179  {
180  constexpr int Degree = 4;
181  const int order = 2 * (Degree - 1);
182 
183  lagrange::enable_fpe();
184  auto p1 = triangle.row(0);
185  auto p2 = triangle.row(1);
186  auto p3 = triangle.row(2);
187  Eigen::Matrix<T, 3, 1> P1 = get(p1.x(), p1.y());
188  Eigen::Matrix<T, 3, 1> P2 = get(p2.x(), p2.y());
189  Eigen::Matrix<T, 3, 1> P3 = get(p3.x(), p3.y());
190 
191  Eigen::AlignedBox2d bbox;
192  Eigen::Matrix<double, 3, 2, 1> triangle_double;
193  for (auto i = 0; i < 3; i++) {
194  Eigen::Vector2d p_double;
195  p_double << get_value(triangle(i, 0)), get_value(triangle(i, 1));
196  triangle_double.row(i) << p_double(0), p_double(1);
197  bbox.extend(p_double);
198  }
199  const auto bbox_min = bbox.min();
200  const auto bbox_max = bbox.max();
201  const auto [xx1, yy1] = get_coordinate(bbox_min(0), bbox_min(1));
202  const auto [xx2, yy2] = get_coordinate(bbox_max(0), bbox_max(1));
203  const int num_pixels = std::max(abs(xx2 - xx1), abs(yy2 - yy1)) + 1;
204  assert(num_pixels > 0);
205  const double pixel_size = bbox.diagonal().maxCoeff() / num_pixels;
206 
207  auto vol = [](const auto& a, const auto& b) -> T {
208  return a.x() * b.y() - b.x() * a.y();
209 
210  // return d1.homogeneous().cross(d2.homogeneous()).z();
211  };
212  // calculate the barycentric coordinate of the a point using u, v cooridnates
213  // returns the 3d coordinate on the current mesh
214  auto get_p_tri = [&](const auto& p) -> Eigen::Matrix<T, 3, 1> {
215  /*
216  λ1 = ((v2 - v3)(u - u3) + (u3 - u2)(v - v3)) / ((v2 - v3)(u1 - u3) + (u3 - u2)(v1 - v3))
217  λ2 = ((v3 - v1)(u - u3) + (u1 - u3)(v - v3)) / ((v2 - v3)(u1 - u3) + (u3 - u2)(v1 - v3))
218  λ3 = 1 - λ1 - λ2
219  z = λ1 * z1 + λ2 * z2 + λ3 * z3
220  */
221  auto d23 = p2 - p3;
222  auto d13 = p1 - p3;
223  auto dp3 = p - p3;
224 
225  T v = vol(d13, d23);
226 
227 
228  T lambda1 = vol(dp3, d23) / v;
229  T lambda2 = -vol(dp3, d13) / v;
230  T lambda3 = 1 - (lambda1 + lambda2);
231 
232  Eigen::Matrix<T, 3, 1> p_tri = (lambda1 * P1 + lambda2 * P2 + lambda3 * P3);
233  return p_tri;
234  };
235  auto check_degenerate = [&]() -> bool {
236  auto d23 = p2 - p3;
237  auto d13 = p1 - p3;
238  T v = vol(d13, d23);
239  return v != 0.;
240  };
241 
242  T value = T(0.);
243  auto& cache = m_cache.local();
244  auto& quad = cache.quad;
245  for (auto y = 0; y < num_pixels; ++y) {
246  for (auto x = 0; x < num_pixels; ++x) {
247  Eigen::AlignedBox2d box;
248  box.extend(bbox.min() + Eigen::Vector2d(x * pixel_size, y * pixel_size));
249  box.extend(
250  bbox.min() + Eigen::Vector2d((x + 1) * pixel_size, (y + 1) * pixel_size));
251  wmtk::ClippedQuadrature rules;
252  rules
253  .clipped_triangle_box_quadrature(order, triangle_double, box, quad, &cache.tmp);
254  for (size_t i = 0; i < quad.size(); ++i) {
255  auto p = quad.points().row(i);
256  auto pT = p.template cast<T>();
257  if (!check_degenerate())
258  continue;
259  else {
260  Eigen::Matrix<T, 3, 1> tmpp_displaced = get(pT.x(), pT.y());
261  Eigen::Matrix<T, 3, 1> tmpp_tri = get_p_tri(pT);
262  Eigen::Matrix<T, 3, 1> diffp = tmpp_displaced - tmpp_tri;
263  value += diffp.squaredNorm() * T(quad.weights()(i));
264  }
265  }
266  }
267  }
268  value = value * triangle_3d_area<T>(P1, P2, P3);
269  value = value / triangle_2d_area<T>(p1.transpose(), p2.transpose(), p3.transpose());
270 
271  return value;
272  }
