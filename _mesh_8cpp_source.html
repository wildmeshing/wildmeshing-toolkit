33 std::vector<Tuple> Mesh::get_all(PrimitiveType type, const bool include_deleted) const
34 {
35  std::vector<Tuple> ret;
36 
37  if (static_cast<int8_t>(type) > top_cell_dimension()) return ret;
38 
39  const int64_t cap = capacity(type);
40 
41  ConstAccessor<char> flag_accessor = get_flag_accessor(type);
42  const attribute::CachingAccessor<char>& flag_accessor_indices = flag_accessor.index_access();
43  ret.reserve(cap);
44  for (size_t index = 0; index < cap; ++index) {
45  if (flag_accessor_indices.const_scalar_attribute(index) & 1)
46  ret.emplace_back(tuple_from_id(type, index));
47  else if (include_deleted)
48  ret.emplace_back();
49  }
50  return ret;
51 }
