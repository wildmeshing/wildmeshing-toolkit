155 bool SimplicialComplex::link_cond_bd_2d(const Mesh& m, Tuple t)
156 {
157  assert(m.top_cell_dimension() == 2);
158  // step1 check normal link condition
159  if (!link_cond(m, t)) {
160  return false;
161  }
162  // check if dummy vertex w is included in the lhs
163

165  Simplex input_v(PrimitiveType::Vertex, _v);
166  std::vector<Tuple> ret;
167  // get one_ring_edges from open_star
168  auto one_ring_edges = open_star(m, input_v).get_simplices(PrimitiveType::Edge);
169  for (const auto& _e : one_ring_edges) {
170  if (m.is_boundary(_e.tuple(), PrimitiveType::Edge)) {
171  if (simplex::utils::SimplexComparisons::equal(
172  m,
173  Simplex(PrimitiveType::Vertex, _e.tuple()),
174  input_v)) {
175  ret.push_back(m.switch_tuple(_e.tuple(), PrimitiveType::Vertex));
176  } else {
177  ret.push_back(_e.tuple());
178  }
179  }
180  }
181  return ret;
182  };
183  // case 1: edge ab is a boundary edge, in this case dummy vertex w is in lnk(ab), need to check
184  // if there are any common edges connected with w in lnk_w^0(a)âˆ©lnk_w^0(b)
185  auto bd_neighbors_a = get_bd_edges(t);
186  auto bd_neighbors_b = get_bd_edges(m.switch_tuple(t, PrimitiveType::Vertex));
187  if (m.is_boundary(t, PrimitiveType::Edge)) {
188  assert(bd_neighbors_a.size() == 2); // if guarantee 2-manifold
189  assert(bd_neighbors_b.size() == 2); // if guarantee 2-manifold
190  for (auto e_a : bd_neighbors_a) {
191  for (auto e_b : bd_neighbors_b) {
192  if (simplex::utils::SimplexComparisons::equal(
193  m,
194  Simplex(PrimitiveType::Vertex, e_a),
195  Simplex(PrimitiveType::Vertex, e_b))) {
196  // find common edge, link condition fails
197  return false;
198  }
199  }
200  }
201  } else {
202  if (bd_neighbors_a.size() == 0 || bd_neighbors_b.size() == 0) {
203  // in this case, lnk_w^0(a) âˆ© lnk_w^0(b) == lnk(a) âˆ© lnk(b) == lnk(ab) == lnk_w^0(ab)
204  return true;
205  } else {

207  return false;
208  }

231 bool SimplicialComplex::edge_collapse_possible_2d(const TriMesh& m, const Tuple& t)
232 {
233  // cannot collapse edges connecting two boundaries unless the edge itself is a boundary
234  if (m.is_boundary_vertex(t) && m.is_boundary_vertex(m.switch_vertex(t)) &&
235  !m.is_boundary_edge(t)) {
236  return false;
237  }
238 
239  auto opp = [&m](const Tuple& tlambda) { return m.switch_vertex(m.switch_face(tlambda)); };
240 
241  const Tuple h0 = m.next_edge(t);
242  const Tuple h0_next = m.next_edge(h0);
243 
244  if (!m.is_boundary_edge(h0)) {
245  const Tuple h0_opp = opp(h0);
246  if (h0_opp == h0_next) {
247  return false;
248  }
249  }
250 
251  // valence 1 check
252  if (!m.is_boundary_edge(t)) {
253  const Tuple t_opp = opp(t);
254  const Tuple h1 = m.next_edge(t_opp);
255  const Tuple h1_next = m.next_edge(h1);
256  if (h0 == t_opp && h1 == t) {
257  return false;
258  }
259  if (!m.is_boundary_edge(h1)) {
260  const Tuple h1_opp = opp(h1);
261  if (h1_opp == h1_next) {
262  return false;
263  }
264  }
265  }

276 std::vector<Simplex> SimplicialComplex::vertex_one_ring(const TriMesh& m, Tuple t)
277 {
278  std::vector<Simplex> one_ring;
279  one_ring.reserve(6);
280 
281  Tuple iter = t;
282  do {
283  one_ring.emplace_back(Simplex::vertex(m.switch_vertex(iter)));
284  if (m.is_boundary_edge(iter)) {
285  break;
286  }
287  iter = m.switch_edge(m.switch_face(iter));
288  } while (iter != t);
289 
290  // in case of a boundary, collect the other side too
291  if (iter != t || m.is_boundary_edge(t)) {
292  iter = m.switch_edge(t);
293  do {
294  one_ring.emplace_back(Simplex::vertex(m.switch_vertex(iter)));
295  if (m.is_boundary_edge(iter)) {
296  break;
297  }
298  iter = m.switch_edge(m.switch_face(iter));
299  } while (iter != t);
300  }
