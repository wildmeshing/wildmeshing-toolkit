3// A local copy of libigl's winding-number evaluation, but with the per-query
4// parallelism driven by wmtk's own threading framework (parallel_for)
5// instead of igl::parallel_for. This means the winding number honours the
6// requested `num_threads` (like every other parallel section in wmtk) rather than
7// always grabbing all hardware cores.
8//
9// The actual algorithm is unchanged: it reuses igl::WindingNumberAABB (a
10// header-only hierarchical accelerator) so it stays as fast as igl::winding_number
11// while producing identical results.

33inline void winding_number(
34 const Eigen::MatrixXd& V,
35 const Eigen::MatrixXi& F,
36 const Eigen::MatrixXd& O,
37 Eigen::VectorXd& W,
38 int num_threads)
39{
40 W.setZero(O.rows());
41 if (O.rows() == 0 || F.rows() == 0 || V.rows() == 0) return;
42
43 // Build the accelerator once (same as igl::winding_number for triangle meshes).
44 igl::WindingNumberAABB<Eigen::Matrix<double, 1, 3>, Eigen::MatrixXd, Eigen::MatrixXi> hier(
45 V,
46 F);
47 hier.grow();
48
49 // hier.winding_number(p) is const and used by igl the same way from parallel_for,
50 // so concurrent queries against the shared hierarchy are safe.
51 threading::parallel_for(
52 threading::range(0, static_cast<size_t>(O.rows())),
53 [&](const threading::range& r) {
54 for (int o = r.begin(); o < r.end(); ++o) {
55 W(o) = hier.winding_number(O.row(o));
56 }
57 },
58 std::max(num_threads, 1));
59}

61} // namespace wmtk::utils
