251 void MshReader::extract_simplex_elements()
252 {
253  const auto* vertex_block = get_vertex_block<DIM>();
254  const auto* element_block = get_simplex_element_block<DIM>();
255  if (element_block == nullptr) return;
256  assert(vertex_block != nullptr);
257 
258  const size_t num_elements = element_block->num_elements_in_block;
259  if (num_elements == 0) return;
260  assert(vertex_block->num_nodes_in_block != 0);
261 
262  const size_t vert_tag_offset = vertex_block->tags.front();
263  const size_t elem_tag_offset = element_block->data.front();
264  for (size_t i = 0; i < num_elements; i++) {
265  const size_t tag = element_block->data[i * (DIM + 2)] - elem_tag_offset;
266  assert(tag < num_elements);
267  const auto* element = element_block->data.data() + i * (DIM + 2) + 1;
268 
269  if constexpr (DIM == 1) {

271  } else if constexpr (DIM == 2) {
