14 template <typename T>
15 std::tuple<size_t, size_t, T> sample_coord(const T coord, const size_t size)
16 {
17  const double xcoord = get_value(coord);
18  assert(0.f <= xcoord && xcoord <= static_cast<double>(size));
19  size_t coord0, coord1;
20  T t(0.0);
21  if (xcoord <= 0.5f) {
22  coord0 = 0;
23  coord1 = 0;
24  t = T(0.5) + coord;
25  } else if (xcoord + 0.5f >= static_cast<double>(size)) {
26  coord0 = size - 1;
27  coord1 = size - 1;
28  t = coord - (T(coord0) + T(0.5));
29  } else {
30  assert(1 < size);
31  coord0 = std::min(size - 2, static_cast<size_t>(xcoord - 0.5f));
32  coord1 = coord0 + 1;
33  t = coord - (T(coord0) + T(0.5));
34  }
35  return std::make_tuple(coord0, coord1, t);
36 };

38 inline uint32_t to_morton_z_order(uint16_t x, uint16_t y)
39 {
40  constexpr auto swizzle = [](uint16_t x) {
41  uint32_t z = x; // x &= 0x00'00'FF'FF
42  z = (z ^ (z << 8)) & 0x00'FF'00'FF;
43  z = (z ^ (z << 4)) & 0x0F'0F'0F'0F;
44  z = (z ^ (z << 2)) & 0x33'33'33'33;
45  z = (z ^ (z << 1)) & 0x55'55'55'55;
46  return z;
47  };
48  auto z = (swizzle(y) << 1) + swizzle(x);
49  return z;
50 }

252 inline bool point_in_triangle(
253  const Eigen::Matrix<double, 3, 2, Eigen::RowMajor>& triangle,
254  const Eigen::Vector2d& point)
255 {
256  const auto& a = triangle.row(0);
257  const auto& b = triangle.row(1);
258  const auto& c = triangle.row(2);
259 
260  const double as_x = point.x() - a.x();
261  const double as_y = point.y() - a.y();
262 
263  bool s_ab = (b.x() - a.x()) * as_y - (b.y() - a.y()) * as_x > 0;
264 
265  if ((c.x() - a.x()) * as_y - (c.y() - a.y()) * as_x > 0 == s_ab) {
266  return false;
267  }
268  if ((c.x() - b.x()) * (point.y() - b.y()) - (c.y() - b.y()) * (point.x() - b.x()) > 0 != s_ab) {
269  return false;
270  }
271  return true;
272 }
