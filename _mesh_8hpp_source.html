649 template <std::forward_iterator ContainerType>
650 #else
651 template <typename ContainerType>
652 #endif
653 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
654 {
655  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
656  Tuple r = tuple;
657  const PrimitiveType top_type = top_simplex_type();
658  for (const PrimitiveType primitive : sequence) {
659  // for top level simplices we cannot navigate across boundaries
660  if (primitive == top_type && is_boundary(r)) {
661  assert(!is_boundary(r));
662  r = {};
663  return r;
664  }
665  r = switch_tuple(r, primitive);
666  }
667  return r;
668 }
