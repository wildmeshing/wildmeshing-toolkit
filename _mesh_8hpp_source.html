875 #if defined(__cpp_concepts)
876 template <std::forward_iterator ContainerType>
877 #else
878 template <typename ContainerType>
879 #endif
880 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
881 {
882  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
883  Tuple r = tuple;
884  const PrimitiveType top_type = top_simplex_type();
885  for (const PrimitiveType primitive : sequence) {
886  // for top level simplices we cannot navigate across boundaries
887  if (primitive == top_type && is_boundary(r)) {
888  assert(!is_boundary(r));
889  r = {};
890  return r;
891  }
892  r = switch_tuple(r, primitive);
893  }
894  return r;
895 }
