154 template <int DIM>
155 void MshReader::extract_simplex_elements()
156 {
157  const mshio::NodeBlock* vertex_block = get_vertex_block(DIM);
158  const mshio::ElementBlock* element_block = get_simplex_element_block(DIM);
159  if (element_block == nullptr) return;
160  assert(vertex_block != nullptr);
161 
162  const size_t num_elements = element_block->num_elements_in_block;
163  if (num_elements == 0) return;
164  assert(vertex_block->num_nodes_in_block != 0);
165 
166  S.resize(num_elements, DIM + 1);
167 
168  const size_t vert_tag_offset = vertex_block->tags.front();
169  const size_t elem_tag_offset = element_block->data.front();
170  for (size_t i = 0; i < num_elements; i++) {
171  const size_t tag = element_block->data[i * (DIM + 2)] - elem_tag_offset;
172  assert(tag < num_elements);
173  const size_t* element = element_block->data.data() + i * (DIM + 2) + 1;
174 
175  using Vector = wmtk::Vector<size_t, DIM + 1>;
176  using ConstMapType = typename Vector::ConstMapType;
177 
178  ConstMapType element_vec(element);
179  const auto vertex_tag_offset_vec = Vector::Constant(vert_tag_offset);
180 
181  S.row(tag) = (element_vec - vertex_tag_offset_vec).transpose().template cast<int64_t>();
182  }

271 void MshReader::extract_element_attribute(
272  Mesh& m,
273  const std::string& attr_name,
274  PrimitiveType primitive_type)
275 {
276  assert(primitive_type == m.top_simplex_type());
277  const int64_t primitive_dimension = get_primitive_type_id(primitive_type);
278  const mshio::ElementBlock* element_block = get_simplex_element_block(primitive_dimension);
279  const auto tuples = m.get_all(m.top_simplex_type());
280  const size_t tag_offset = element_block->data.front();
281 
282  for (const auto& data : m_spec.element_data) {
283  if (data.header.string_tags.front() != attr_name) {
284  continue;
285  }
286  assert(data.entries.size() == tuples.size());
287 
288  const int64_t attr_dim = data.header.int_tags[1];
289 
290  auto tag_handle = m.register_attribute<double>(attr_name, primitive_type, attr_dim);
291  auto acc = m.create_accessor<double>(tag_handle);
292 
293  for (size_t i = 0; i < tuples.size(); ++i) {
294  const auto& entry = data.entries[i];
295  const size_t tag = entry.tag - tag_offset;
296  assert(tag < element_block->num_elements_in_block);
297  assert(tag == i);
298  if (attr_dim == 1) {
299  acc.scalar_attribute(tuples[i]) = entry.data[0];
300  } else {
301  acc.vector_attribute(tuples[i]) =
302  Eigen::Map<const Eigen::VectorXd>(entry.data.data(), entry.data.size());
303  }
304  }
305  return;
306  }

309 auto MshReader::generate(const std::optional<std::vector<std::vector<std::string>>>&
310  extra_attributes_opt) -> std::shared_ptr<Mesh>
311 {
312  std::shared_ptr<Mesh> res;
313  switch (get_mesh_dimension()) {
314  case 3: res = generateT<3>(); break;
315  case 2: res = generateT<2>(); break;
316  case 1: res = generateT<1>(); break;
317  default:
318  case 0: res = std::make_shared<PointMesh>();
319  }
320  assert(V.size() > 0);
321 
322  if (extra_attributes_opt.has_value()) {
323  const auto& extra_attributes = extra_attributes_opt.value();
324  // save all valid scalar attributes on tets
325  std::set<std::string> valid_attr_names;
326  for (const auto& data : m_spec.element_data) {
327  // validness: int_tags[2] stores the dimension of an element_data, which should
328  // match the size of tets
329  if (data.header.int_tags[2] != get_num_simplex_elements(3)) {
330  continue;
331  }
332  valid_attr_names.insert(data.header.string_tags.front());
333  }
334  for (size_t dim = 0; dim < extra_attributes.size(); ++dim) {
335  const auto& attr_names = extra_attributes[dim];
336  for (const std::string& attr : attr_names) {
337  if (valid_attr_names.count(attr) == 0) {
338  log_and_throw_error("Attribute " + attr + " does not exist in the msh file.");
339  }
340  extract_element_attribute(*res, attr, get_primitive_type_from_id(dim));
341  }
342  }
343  }
344  mesh_utils::set_matrix_attribute(V, "vertices", PrimitiveType::Vertex, *res);
345  return res;
346 }
