86  // Edge length update
87  auto compute_edge_length = [](const Eigen::MatrixXd& P) -> Eigen::VectorXd {
88  assert(P.cols() == 2);
89  assert(P.rows() == 2 || P.rows() == 3);
90  return Eigen::VectorXd::Constant(1, (P.col(0) - P.col(1)).norm());
91  };

93  std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<double, double>>(
94  edge_length_attribute,
95  pt_attribute,
96  compute_edge_length);
97 
99  // computing edge lengths
100  const auto edges = mesh->get_all(PrimitiveType::Edge);
101  for (const auto& e : edges) {
102  const auto p0 = pt_accessor.vector_attribute(e);
103  const auto p1 = pt_accessor.vector_attribute(mesh->switch_vertex(e));
104 
105  edge_length_accessor.scalar_attribute(e) = (p0 - p1).norm();
106  }

109  // computng bbox diagonal
110  Eigen::VectorXd bmin(options.planar ? 2 : 3);
111  bmin.setConstant(std::numeric_limits<double>::max());
112  Eigen::VectorXd bmax(options.planar ? 2 : 3);
113  bmax.setConstant(std::numeric_limits<double>::min());
114 
115  const auto vertices = mesh->get_all(PrimitiveType::Vertex);
116  for (const auto& v : vertices) {
117  const auto p = pt_accessor.vector_attribute(v);
118  for (int64_t d = 0; d < bmax.size(); ++d) {
119  bmin[d] = std::min(bmin[d], p[d]);
120  bmax[d] = std::max(bmax[d], p[d]);
121  }
122  }
123

132  };

134  assert(s.primitive_type() == PrimitiveType::Edge);
