45  int64_t max_size = flag_accessor.reserved_size();

190 std::tuple<std::vector<std::vector<int64_t>>, std::vector<std::vector<int64_t>>> Mesh::consolidate()
191 {
192  // Number of dimensions
193  int64_t tcp = top_cell_dimension() + 1;
194 
195  // Store the map from new indices to old. First index is dimensions, second simplex id
196  std::vector<std::vector<int64_t>> new2old(tcp);
197  // Store the map from old indices to new. First index is dimensions, second simplex id
198  std::vector<std::vector<int64_t>> old2new(tcp);
199 
200  // Initialize both maps
201  for (int64_t d = 0; d < tcp; d++) {
202  attribute::Accessor<char> flag_accessor = get_flag_accessor(wmtk::get_primitive_type_from_id(d));
203  for (int64_t i = 0; i < capacity(wmtk::get_primitive_type_from_id(d)); ++i) {
204  if (flag_accessor.index_access().scalar_attribute(i) & 1) {
205  old2new[d].push_back(new2old[d].size());
206  new2old[d].push_back(old2new[d].size() - 1); // -1 since we just pushed into it
207  } else {
208  old2new[d].push_back(-1);
209  }
210  }
211  }
212 
213  // Use new2oldmap to compact all attributes
214  auto run = [&](auto&& mesh_attrs) {
215  for (int64_t d = 0; d < mesh_attrs.size(); ++d) {
216  mesh_attrs[d].reserve(new2old.size());
217  for (auto& h : mesh_attrs[d].m_attributes) {
218  h->consolidate(new2old[d]);
219  }
220  }
221  };
222  run(m_attribute_manager.m_char_attributes);
223  run(m_attribute_manager.m_long_attributes);
224 
225  run(m_attribute_manager.m_double_attributes);
226 
227  run(m_attribute_manager.m_rational_attributes);
228 
229  // Update the attribute size in the manager
230  for (int64_t d = 0; d < tcp; d++) {
231  m_attribute_manager.m_capacities[d] = new2old[d].size();
232  }
233 
234  // Apply old2new to attributes containing indices
235  std::vector<std::vector<TypedAttributeHandle<int64_t>>> handle_indices =
236  connectivity_attributes();
237 
238  for (int64_t d = 0; d < tcp; d++) {
239  for (int64_t i = 0; i < handle_indices[d].size(); ++i) {
240  attribute::Accessor<int64_t> accessor = create_accessor<int64_t>(handle_indices[d][i]);
241  accessor.attribute().index_remap(old2new[d]);
242  }
243  }
244 
245  {
246  constexpr static int64_t TUPLE_SIZE = multimesh::utils::TUPLE_SIZE; // in terms of int64_t
247  constexpr static int64_t GLOBAL_ID_INDEX = multimesh::utils::GLOBAL_ID_INDEX;
248  const static std::vector<Eigen::Index> image_map_offsets{
249  Eigen::Index(TUPLE_SIZE + GLOBAL_ID_INDEX)};
250  const static std::vector<Eigen::Index> domain_map_offsets{Eigen::Index(GLOBAL_ID_INDEX)};
251  size_t dim = get_primitive_type_id(top_simplex_type());
252  const auto& top_map = old2new[dim];
253  if (auto parent_ptr = m_multi_mesh_manager.m_parent; parent_ptr != nullptr) {
254  {
255  int64_t child_id = m_multi_mesh_manager.m_child_id;
256  const auto& child_data = parent_ptr->m_multi_mesh_manager.m_children[child_id];
257  const auto handle = child_data.map_handle;
258  auto acc = parent_ptr->create_accessor(handle);
259  auto& attr = acc.attribute();
260  attr.index_remap(top_map, image_map_offsets);
261  }
262 
263  {
264  const auto handle = m_multi_mesh_manager.map_to_parent_handle;
265  auto acc = create_accessor(handle);
266  auto& attr = acc.attribute();
267  attr.index_remap(top_map, domain_map_offsets);
268  }
269  }
270 
271  for (const auto& child_data : m_multi_mesh_manager.m_children) {
272  {
273  const auto handle = child_data.map_handle;
274  auto acc = create_accessor(handle);
275  auto& attr = acc.attribute();
276  attr.index_remap(top_map, domain_map_offsets);
277  }
278  {
279  const auto handle = child_data.mesh->m_multi_mesh_manager.map_to_parent_handle;
280  auto acc = child_data.mesh->create_accessor(handle);
281  auto& attr = acc.attribute();
282  attr.index_remap(top_map, image_map_offsets);
283  }
284  }
285  }
286 
287  // Return both maps for custom attribute remapping
288  return {new2old, old2new};
