226 std::vector<std::shared_ptr<Mesh>> Mesh::get_all_child_meshes() const
227 {
228  std::vector<std::shared_ptr<Mesh>> children;
229  auto direct_children = get_child_meshes();
230  for (const auto& child : direct_children) {
231  children.emplace_back(child);
232  auto child_children = child->get_all_child_meshes();
233  if (child_children.empty()) {
234  continue;
235  }
236  std::copy(child_children.begin(), child_children.end(), std::back_inserter(children));
237  }
