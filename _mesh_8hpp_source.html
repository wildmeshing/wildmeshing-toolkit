943 {
944  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
945  Tuple r = tuple;

947 
948  const int64_t boundary_dim = top_cell_dimension() - 1;
949  const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
950

952  // for top level simplices we cannot navigate across boundaries
953  if (primitive == top_type && is_boundary(boundary_pt, r)) {
954  assert(!is_boundary(boundary_pt, r));
955  r = {};
956  return r;
957  }
958  r = switch_tuple(r, primitive);
959  }
