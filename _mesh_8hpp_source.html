913 template <std::forward_iterator ContainerType>
914 #else
915 template <typename ContainerType>
916 #endif
917 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
918 {
919  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
920  Tuple r = tuple;
921  const PrimitiveType top_type = top_simplex_type();
922 
923  const int64_t boundary_dim = top_cell_dimension() - 1;
924  const PrimitiveType boundary_pt = static_cast<PrimitiveType>(boundary_dim);
925 
926  for (const PrimitiveType primitive : sequence) {
927  // for top level simplices we cannot navigate across boundaries
928  if (primitive == top_type && is_boundary(r, boundary_pt)) {
929  assert(!is_boundary(r, boundary_pt));
930  r = {};
931  return r;
932  }
933  r = switch_tuple(r, primitive);
934  }
935  return r;
936 }
