124  /* @brief runs the node functor on every node in the subgraph
125  * @param mesh the mesh whose subgraph will be run
126  * @param simplex the simplex whose subgraph will be run
127  * */
128  template <typename MeshType_>
129  void run(MeshType_&& current_mesh)
130  {
131  using MeshType = std::decay_t<MeshType_>;
132 
133 
134  // short circuit operations that happen below the desired dimension
135  constexpr static long MeshDim = wmtk::utils::metaprogramming::cell_dimension_v<MeshType>;
136 
137 
138  // pre-compute all of the child tuples in case the node functor changes the mesh that
139  // breaks the traversal down
140  auto& child_datas = current_mesh.m_multi_mesh_manager.children();
141 
142  // go over each child mesh / child simplices and run the node functor on them
143  // then recurses this function onto the children
144  for (size_t child_index = 0; child_index < child_datas.size(); ++child_index) {
145  auto&& child_data = child_datas[child_index];
146  Mesh& child_mesh_base = *child_data.mesh;
147 
148 
149  auto child_mesh_variant =
150  wmtk::utils::metaprogramming::as_mesh_variant(child_mesh_base);
151  std::visit(
152  [&](auto&& child_mesh_) noexcept {
153  auto&& child_mesh = child_mesh_.get();
154  using ChildType = std::decay_t<decltype(child_mesh)>;
155 
156  constexpr static long ChildDim =
157  wmtk::utils::metaprogramming::cell_dimension_v<ChildType>;
158 
159  // std::visit compiles all combinations of meshes, and
160  // the behavior is undefined if MeshDim < ChildDim.
161  //
162  // we assert this to make sure the code is correct at
163  // runtime, we if constexpr after to make sure the
164  // compiler doesn't try to compile code in these cases
165  assert(MeshDim >= ChildDim);
166 
167  if constexpr (MeshDim >= ChildDim) {
168  run(child_mesh);
169  }
170  },
171  child_mesh_variant);
172  }
173 
174  visitor.m_node_functor(current_mesh);
175  }
