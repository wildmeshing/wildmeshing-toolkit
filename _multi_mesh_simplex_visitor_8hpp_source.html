181  * @param mesh the mesh whose subgraph will be run

183  * */
184  template <typename MeshType_>
185  void run(MeshType_&& current_mesh, const simplex::Simplex& simplex)
186  {
187  assert(current_mesh.is_valid_slow(simplex.tuple()));
188  using MeshType = std::decay_t<MeshType_>;

190 
191  // short circuit operations that happen below the desired dimension
192  constexpr static long MeshDim = wmtk::utils::metaprogramming::cell_dimension_v<MeshType>;
193  if constexpr (cell_dimension > MeshDim) {
194  return;
195  }
196 #if defined(WMTK_MESH_VISITOR_ONLY_SUPPORTS_NONCONST_REFERENCE)
197  using CurReturnType = GetReturnType_t<MeshType>;
198 #else
199  constexpr static bool CurIsConst = std::is_const_v<MeshType>;
200  using CurReturnType = GetReturnType_t<CurIsConst, MeshType>;
201 #endif
202 
203  constexpr static bool CurHasReturn = !std::is_void_v<CurReturnType>;
204 
205  // pre-compute all of the child tuples in case the node functor changes the mesh that
206  // breaks the traversal down
207  auto& child_datas = current_mesh.m_multi_mesh_manager.children();
208  std::vector<std::vector<Simplex>> mapped_child_simplices;
209  mapped_child_simplices.reserve(child_datas.size());
210 
211 
212  // in-place convert this tuple into all of the child simplices
213  // TODO: this repeatedly extracts every version of hte input smiplex,
214  // we could cache this in the future
215  std::transform(
216  child_datas.begin(),
217  child_datas.end(),
218  std::back_inserter(mapped_child_simplices),
219  [&](const auto& child_data) {
220  Mesh& child_mesh = *child_data.mesh;
221 
222  auto r = current_mesh.map_to_child(child_mesh, simplex);
223 #if !defined(NDEBUG)
224  for (const auto& s : r) {
225  assert(child_mesh.is_valid_slow(s.tuple()));
226  }
227 #endif
228 
229  return r;
230  });
231 
232 
233  // go over each child mesh / child simplices and run the node functor on them
234  // then recurses this function onto the children
235  for (size_t child_index = 0; child_index < child_datas.size(); ++child_index) {
236  auto&& child_data = child_datas[child_index];
237  auto&& simplices = mapped_child_simplices[child_index];
238  Mesh& child_mesh_base = *child_data.mesh;
239 
240 #if !defined(NDEBUG)
241  for (const auto& s : simplices) {
242  assert(child_mesh_base.is_valid_slow(s.tuple()));
243  }
244 #endif

246  auto child_mesh_variant =
247  wmtk::utils::metaprogramming::as_mesh_variant(child_mesh_base);
248  std::visit(

250  auto&& child_mesh = child_mesh_.get();
251  using ChildType = std::decay_t<decltype(child_mesh)>;
252 #if defined(WMTK_MESH_VISITOR_ONLY_SUPPORTS_NONCONST_REFERENCE)
253  using ChildReturnType = GetReturnType_t<ChildType>;
254 #else
255  constexpr static bool ChildIsConst = std::is_const_v<ChildType>;
256  using ChildReturnType = GetReturnType_t<ChildIsConst, ChildType>;
257 #endif

259  constexpr static bool ChildHasReturn = !std::is_void_v<ChildReturnType>;
260  constexpr static long ChildDim =
261  wmtk::utils::metaprogramming::cell_dimension_v<ChildType>;
262 
263  // std::visit compiles all combinations of meshes, and
264  // the behavior is undefined if MeshDim < ChildDim.
265  //
266  // we assert this to make sure the code is correct at
267  // runtime, we if constexpr after to make sure the
268  // compiler doesn't try to compile code in these cases
269  assert(MeshDim >= ChildDim);
270 
271  if constexpr (MeshDim >= ChildDim) {
272  for (const simplex::Simplex& child_simplex : simplices) {
273 #if !defined(NDEBUG)
274  assert(child_mesh.is_valid_slow(child_simplex.tuple()));
275 #endif
276  run(child_mesh, child_simplex);
277 
278  if constexpr (HasReturnCache && ChildHasReturn && CurHasReturn) {
279  auto parent_id = m_return_data.get_id(current_mesh, simplex);
280  auto child_id = m_return_data.get_id(child_mesh, child_simplex);
281  // spdlog::info(
282  // "MultiMeshSimplexVisitor[{}=>{}] adding to edges edge simplex {}
283  // " "child " "simplex{}",
284  // fmt::join(current_mesh.absolute_multi_mesh_id(), ","),
285  // fmt::join(child_mesh.absolute_multi_mesh_id(), ","),
286  // wmtk::utils::TupleInspector::as_string(
287  // std::get<1>(parent_id).tuple()),
288  // wmtk::utils::TupleInspector::as_string(
289  // std::get<1>(child_id).tuple()));
290  edge_events.emplace_back(parent_id, child_id);
291  }
292  }
293  }
294  },
295  child_mesh_variant);
296  }
297 
298  // after running on the chlidren, we finally run the operator and record the return data
299  if constexpr (CurHasReturn) {
300  auto current_return = visitor.m_node_functor(current_mesh, simplex);
301 
302  m_return_data.add(current_return, current_mesh, simplex);
303  } else {
304  visitor.m_node_functor(current_mesh, simplex);
305  }
306  }
