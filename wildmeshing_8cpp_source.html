76  // Storing edge lengths

79  auto edge_length_accessor = mesh->create_accessor(edge_length_attribute);

84  auto pt_attribute = mesh->get_attribute_handle<double>("vertices", PrimitiveType::Vertex);
85  auto pt_accessor = mesh->create_accessor(pt_attribute);
86 
87  // Edge length update
88  auto compute_edge_length = [](const Eigen::MatrixXd& P) -> Eigen::VectorXd {
89  assert(P.cols() == 2);
90  assert(P.rows() == 2 || P.rows() == 3);
91  return Eigen::VectorXd::Constant(1, (P.col(0) - P.col(1)).norm());
92  };
93  auto edge_length_update =
94  std::make_shared<wmtk::operations::SingleAttributeTransferStrategy<double, double>>(
95  edge_length_attribute,
96  pt_attribute,
97  compute_edge_length);
98 
100  // computing edge lengths
101  const auto edges = mesh->get_all(PrimitiveType::Edge);
102  for (const auto& e : edges) {
103  const auto p0 = pt_accessor.vector_attribute(e);
104  const auto p1 = pt_accessor.vector_attribute(mesh->switch_vertex(e));
105 
106  edge_length_accessor.scalar_attribute(e) = (p0 - p1).norm();
107  }
108 
110  // computng bbox diagonal
111  Eigen::VectorXd bmin(options.planar ? 2 : 3);
112  bmin.setConstant(std::numeric_limits<double>::max());
113  Eigen::VectorXd bmax(options.planar ? 2 : 3);
114  bmax.setConstant(std::numeric_limits<double>::min());
115 
116  const auto vertices = mesh->get_all(PrimitiveType::Vertex);
117  for (const auto& v : vertices) {
118  const auto p = pt_accessor.vector_attribute(v);
119  for (long d = 0; d < bmax.size(); ++d) {
120  bmin[d] = std::min(bmin[d], p[d]);
121  bmax[d] = std::max(bmax[d], p[d]);
122  }
123  }
124 
125  const double bbdiag = (bmax - bmin).norm();
126  const double target_edge_length = options.target_edge_length * bbdiag;
127 
129  // Lambdas for priority
130  auto long_edges_first = [&](const Simplex& s) {
131  assert(s.primitive_type() == PrimitiveType::Edge);
132  return std::vector<double>({-edge_length_accessor.scalar_attribute(s.tuple())});
133  };
134  auto short_edges_first = [&](const Simplex& s) {
135  assert(s.primitive_type() == PrimitiveType::Edge);
136  return std::vector<double>({edge_length_accessor.scalar_attribute(s.tuple())});
137  };
138

143 
144

158  4.0 / 3.0 * target_edge_length));

174  collapse->add_invariant(std::make_shared<TodoSmallerInvariant>(
175  *mesh,
176  edge_length_attribute,
177  4.0 / 5.0 * target_edge_length));
178  collapse->set_priority(short_edges_first);
179

181  tmp->set_standard_collapse_strategy(NewAttributeStrategy::CollapseBasicStrategy::Default);
182  tmp->set_standard_simplex_predicate(NewAttributeStrategy::BasicSimplexPredicate::IsInterior);
183  collapse->set_strategy(pt_attribute, tmp);
184  collapse->set_standard_strategy(edge_length_attribute);

187  ops.emplace_back(collapse);

194  swap->add_invariant(std::make_shared<InteriorEdgeInvariant>(*mesh));

196  swap->add_invariant(std::make_shared<FunctionInvariant>(mesh->top_simplex_type(), amips));
197  swap->set_priority(long_edges_first);
198

200  swap->split().set_standard_strategy(edge_length_attribute);

211  {

213  }
214

217  std::make_shared<function::LocalNeighborsSumFunction>(*mesh, pt_attribute, *amips);
218  ops.emplace_back(std::make_shared<OptimizationSmoothing>(energy));
219  ops.back()->add_invariant(std::make_shared<TriangleInversionInvariant>(*mesh, pt_attribute));
220  ops.back()->add_invariant(std::make_shared<InteriorVertexInvariant>(*mesh));
221  ops.back()->add_transfer_strategy(edge_length_update);
