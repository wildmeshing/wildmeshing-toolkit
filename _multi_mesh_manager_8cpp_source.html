886  std::vector<Tuple> tuples_to_update;

892  tuples_to_update.emplace_back(s.tuple());
893  }
894  break;
895  }
896  case PrimitiveType::Edge: {
897  const auto star_edges = star.get_edges();
898  tuples_to_update.reserve(star_edges.size());
899  for (const Simplex& s : star_edges) {
900  tuples_to_update.emplace_back(s.tuple());
901  }
902  break;
903  }
904  case PrimitiveType::Face: {
905  const auto star_faces = star.get_faces();
906  tuples_to_update.reserve(star_faces.size());
907  for (const Simplex& s : star_faces) {
908  tuples_to_update.emplace_back(s.tuple());
909  }
910  break;
911  }
912  case PrimitiveType::Tetrahedron: {
913  const auto star_tets = star.get_tetrahedra();
914  tuples_to_update.reserve(star_tets.size());
915  for (const Simplex& s : star_tets) {
916  tuples_to_update.emplace_back(s.tuple());
917  }
918  break;
919  }
920  }
921 
922 
923  m.update_cell_hashes(tuples_to_update, hash_accessor);
924

926  auto vertex_new_hash = m.resurrect_tuple(vertex, hash_accessor);
927  const SimplicialComplex star_new_hash =
928  SimplicialComplex::closed_star(m, Simplex::vertex(vertex_new_hash));
