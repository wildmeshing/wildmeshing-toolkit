296 int find(size_t v_id) const
297 {

300 }

303
304 int find_local_edge(size_t v1_id, size_t v2_id) const
305 {
306 std::array<int, 2> e;
307 for (int j = 0; j < 4; j++) {
308 if (v1_id == m_indices[j])
309 e[0] = j;
310 else if (v2_id == m_indices[j])
311 e[1] = j;
312 }
313 if (e[0] > e[1]) std::swap(e[0], e[1]);
314 int i =
315 std::find(m_local_edges.begin(), m_local_edges.end(), e) - m_local_edges.begin();
316 if (i >= m_local_edges.size()) return -1;
317 return i;
318 }
319
320 int find_local_face(size_t v1_id, size_t v2_id, size_t v3_id) const
321 {
322 std::array<int, 3> f;
323 for (int j = 0; j < 4; j++) {
324 if (v1_id == m_indices[j])
325 f[0] = j;
326 else if (v2_id == m_indices[j])
327 f[1] = j;
328 else if (v3_id == m_indices[j])
329 f[2] = j;
330 }
331 std::sort(f.begin(), f.end());
332 int i =
333 std::find(m_local_faces.begin(), m_local_faces.end(), f) - m_local_faces.begin();
334 if (i >= m_local_edges.size()) return -1;
335 return i;
336 }

341 std::tie(r.m_indices, r.m_is_removed, r.hash); // keep the same order
342 }

652 // Atomically reserve `n` contiguous fresh tet/vertex slots. Returns the first
653 // index of the block, or -1 if that would exceed the preallocated capacity (the
654 // caller must then abort the operation before mutating any connectivity).
655 long request_tet_slots(size_t n);

657
658 // Construction/insertion helpers (single-threaded ONLY): guarantee at least
659 // `extra` free tet/vertex slots beyond the current used count, growing the
660 // storage geometrically if necessary. Unlike request_*_slots these never fail;
661 // they are used by the (legacy, non-concurrent) triangle-insertion path where
662 // the mesh is being built rather than edited by concurrent operations.
663 void ensure_free_tet_capacity(size_t extra)
664 {
665 const size_t need = static_cast<size_t>(current_tet_size.load()) + extra;
666 if (m_tet_connectivity.size() >= need) return;
667 const size_t newcap = std::max(need, m_tet_connectivity.size() * 2 + 1);
668 m_tet_connectivity.resize(newcap);
669 if (p_tet_attrs) p_tet_attrs->resize(newcap);
670 if (p_face_attrs) p_face_attrs->resize(4 * newcap);
671 if (p_edge_attrs) p_edge_attrs->resize(6 * newcap);
672 }
673 void ensure_free_vert_capacity(size_t extra)
674 {
675 const size_t need = static_cast<size_t>(current_vert_size.load()) + extra;
676 if (m_vertex_connectivity.size() >= need) return;
677 const size_t newcap = std::max(need, m_vertex_connectivity.size() * 2 + 1);
678 m_vertex_connectivity.resize(newcap);
679 resize_vertex_mutex(newcap);
680 if (p_vertex_attrs) p_vertex_attrs->resize(newcap);
681 }

687 const size_t floor = 64;
