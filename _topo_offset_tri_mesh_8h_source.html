309 void sort_edges_by_length(std::vector<simplex::Edge>& edges)
310 {
311 std::sort(
312 edges.begin(),
313 edges.end(),
314 [this](const simplex::Edge& e1, const simplex::Edge& e2) {
315 double len1 = (m_vertex_attribute[e1.vertices()[0]].m_posf -
316 m_vertex_attribute[e1.vertices()[1]].m_posf)
317 .squaredNorm();
318 double len2 = (m_vertex_attribute[e2.vertices()[0]].m_posf -
319 m_vertex_attribute[e2.vertices()[1]].m_posf)
320 .squaredNorm();
321 return len1 > len2;
322 });
323 }

329 size_t edge_id_from_simplex(const simplex::Edge& e) const

357 std::vector<Tuple> get_edge_adjacent_faces(const Tuple& f) const
358 {
359 std::vector<Tuple> adj_tris;
360 auto tri_1 = f.switch_face(*this);
361 if (tri_1) {
362 adj_tris.push_back(tri_1.value());
363 }
364 auto tri_2 = f.switch_edge(*this).switch_face(*this);
365 if (tri_2) {
366 adj_tris.push_back(tri_2.value());
367 }
368 auto tri_3 = f.switch_vertex(*this).switch_edge(*this).switch_face(*this);
369 if (tri_3) {
370 adj_tris.push_back(tri_3.value());
