228 bool EdgeMesh::is_connectivity_valid() const
229 {
230  // get accessors for topology
231  const attribute::Accessor<int64_t> ev_accessor = create_const_accessor<int64_t>(m_ev_handle);
232  const attribute::Accessor<int64_t> ee_accessor = create_const_accessor<int64_t>(m_ee_handle);
233  const attribute::Accessor<int64_t> ve_accessor = create_const_accessor<int64_t>(m_ve_handle);
234  const attribute::Accessor<char> v_flag_accessor = get_flag_accessor(PrimitiveType::Vertex);
235  const attribute::Accessor<char> e_flag_accessor = get_flag_accessor(PrimitiveType::Edge);
236 
237  // VE and EV
238  for (int64_t i = 0; i < capacity(PrimitiveType::Vertex); ++i) {
239  if (v_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
240  wmtk::logger().debug("Vertex {} is deleted", i);
241  continue;
242  }
243  int cnt = 0;
244  for (int64_t j = 0; j < 2; ++j) {
245  if (ev_accessor.index_access().const_vector_attribute<2>(
246  ve_accessor.index_access().const_scalar_attribute(i))[j] == i) {
247  cnt++;
248  }
249  }
250  if (cnt == 0) {
251  return false;
252  }
253  }
254 
255  // EV and EE
256  for (int64_t i = 0; i < capacity(PrimitiveType::Edge); ++i) {
257  if (e_flag_accessor.index_access().const_scalar_attribute(i) == 0) {
258  wmtk::logger().debug("Edge {} is deleted", i);
259  continue;
260  }
261  // TODO: need to handle cornor case (self-loop)
262  }
263 
264  return true;
265 }
