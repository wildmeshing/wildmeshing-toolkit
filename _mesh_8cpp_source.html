28 std::vector<Tuple> Mesh::get_all(PrimitiveType type, const bool include_deleted) const
29 {
30  ConstAccessor<char> flag_accessor = get_flag_accessor(type);
31  const attribute::CachingAccessor<char>& flag_accessor_indices = flag_accessor.index_access();
32  std::vector<Tuple> ret;
33  int64_t cap = capacity(type);
34  ret.reserve(cap);
35  for (size_t index = 0; index < cap; ++index) {
36  if (flag_accessor_indices.const_scalar_attribute(index) & 1)
37  ret.emplace_back(tuple_from_id(type, index));
38  else if (include_deleted)
39  ret.emplace_back();
40  }
41  return ret;
42 }
