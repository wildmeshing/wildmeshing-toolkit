240  if (ev_accessor.index_access().const_vector_attribute<2>(
241  ve_accessor.index_access().const_scalar_attribute(i))[j] == i) {
242  cnt++;
243  }
244  }
245  if (cnt == 0) {
246  return false;
247  }
248  }
249 
250  // EV and EE
251  for (int64_t i = 0; i < capacity(PrimitiveType::Edge); ++i) {

253  wmtk::logger().debug("Edge {} is deleted", i);
