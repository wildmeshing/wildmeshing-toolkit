137  for (auto& s : dsets) {
138  const std::string dataset = root_dataset + "/" + s;
139 
140 
141  const int64_t stride = hdf5_file.readAttribute<int64_t>(dataset, "stride");
142  const int64_t dimension = hdf5_file.readAttribute<int64_t>(dataset, "dimension");
143  const std::string type = hdf5_file.readAttribute<std::string>(dataset, "type");
144  const std::string name =
145  std::regex_replace(s, std::regex(std::to_string(dimension) + "/"), "");
146 
147  auto pt = PrimitiveType(dimension);
148 
149  if (type == "int64_t") {
150  auto v = hdf5_file.readDataset<std::vector<int64_t>>(dataset);
151  const auto default_val = hdf5_file.readAttribute<int64_t>(dataset, "default_value");
152 
153  set_attribute<int64_t>(default_val, name, pt, stride, v, *mesh);
154  } else if (type == "char") {
155  auto tmp = hdf5_file.readDataset<std::vector<short>>(dataset);
156  const auto default_val = char(hdf5_file.readAttribute<short>(dataset, "default_value"));
157 
158  std::vector<char> v;
159  v.reserve(tmp.size());
160  for (auto val : tmp) v.push_back(char(val));
161 
162 
163  set_attribute<char>(default_val, name, pt, stride, v, *mesh);
164  } else if (type == "double") {
165  auto v = hdf5_file.readDataset<std::vector<double>>(dataset);
166  const auto default_val = hdf5_file.readAttribute<double>(dataset, "default_value");
167 
168 
169  set_attribute<double>(default_val, name, pt, stride, v, *mesh);
170  } else if (type == "rational") {
171  const Eigen::Matrix<int, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> tmp =
172  hdf5_file.readDataset<
173  Eigen::Matrix<int, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>>(dataset);
174  const Eigen::Matrix<char, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> tmp_char =
175  tmp.cast<char>();

178  Rational default_val;

180 
181  std::vector<Rational> v;
182  v.reserve(tmp.rows());
