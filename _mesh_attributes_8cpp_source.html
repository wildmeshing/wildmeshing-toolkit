223

225  remaining_attributes.reserve(attributes.size());
226 
227  std::vector<int64_t> old_to_new_id(m_attributes.size(), -1);
228  for (size_t i = 0, id = 0; i < keep_mask.size(); ++i) {
229  if (keep_mask[i]) {
230  old_to_new_id[i] = id++;
231  remaining_attributes.emplace_back(std::move(m_attributes[i]));
232  // remaining_attributes.emplace_back(std::move(m_attributes[i]));
233  assert(remaining_attributes.size() == id);

235  }

237  // clean up m_handles

239  if (!keep_mask[it->second.index]) {

241  } else {
242  it->second.index = old_to_new_id[it->second.index];
