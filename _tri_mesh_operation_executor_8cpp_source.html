308 {

313  const int64_t f1_top = m_incident_face_datas[0].split_f[1];
314  const int64_t f_old_bottom = m_incident_face_datas[1].fid;
315  const int64_t f0_bottom = m_incident_face_datas[1].split_f[0];
316  const int64_t f1_bottom = m_incident_face_datas[1].split_f[1];
317  auto ff_old_top = ff_accessor.index_access().vector_attribute(f_old_top);
318  auto ff_old_bottom = ff_accessor.index_access().vector_attribute(f_old_bottom);
319  assert(m_mesh.capacity(PrimitiveType::Triangle) > f0_top);
320  assert(m_mesh.capacity(PrimitiveType::Triangle) > f1_top);
321  assert(m_mesh.capacity(PrimitiveType::Triangle) > f0_bottom);
322  assert(m_mesh.capacity(PrimitiveType::Triangle) > f1_bottom);
323 
324  // local edge ids are the same for both, f1 and f2
325  int64_t local_eid_top = -1;
326  int64_t local_eid_bottom = -1;
327  for (size_t i = 0; i < 3; ++i) {
328  if (ff_old_top[i] == f_old_bottom) {
329  local_eid_top = i;
330  }
331  if (ff_old_bottom[i] == f_old_top) {
332  local_eid_bottom = i;
333  }
334  }
335  assert(local_eid_top > -1);
336  assert(local_eid_bottom > -1);
337  // TODO write test for assumming top and bottom new fids are in right correspondence
338  ff_accessor.index_access().vector_attribute(f0_top)[local_eid_top] = f0_bottom;
339  ff_accessor.index_access().vector_attribute(f0_bottom)[local_eid_bottom] = f0_top;
340  ff_accessor.index_access().vector_attribute(f1_top)[local_eid_top] = f1_bottom;

344 void TriMesh::TriMeshOperationExecutor::replace_incident_face(IncidentFaceData& face_data)
345 {
346  // create new faces
347  std::vector<int64_t> new_fids = this->request_simplex_indices(PrimitiveType::Triangle, 2);
348  assert(new_fids.size() == 2);
349 
350  std::copy(new_fids.begin(), new_fids.end(), face_data.split_f.begin());
351 
352  std::vector<int64_t> splitting_edges = this->request_simplex_indices(PrimitiveType::Edge, 1);
353  assert(splitting_edges[0] > -1); // TODO: is this assert reasonable at all?
354  int64_t& split_edge_eid = face_data.new_edge_id;
355  split_edge_eid = splitting_edges[0];
356 
357  // ---------v2--------
358  // | /|\ |
359  // | ef0 / | \ ef1 |
360  // | / | \ |
361  // | / oe \ |
362  // | ee0 | ee1 |
363  // | / | \ |
364  // | / f0 | f1 \ |
365  // | / | \ |
366  // v0--se0-v_new-se1--v1
367 
368  const int64_t v_opp = face_data.opposite_vid; // opposite vertex
369  const int64_t f_old = face_data.fid; // old face
370  auto old_fv = fv_accessor.index_access().const_vector_attribute(f_old).eval();
371  auto old_fe = fe_accessor.index_access().const_vector_attribute(f_old).eval();
372  auto old_ff = ff_accessor.index_access().const_vector_attribute(f_old).eval();
373 
374  // f0
375  for (int j = 0; j < 2; ++j) {
376  int other_index = 1 - j;
377  const int64_t f = face_data.split_f[j]; // new face to insert
378  const auto& ear = face_data.ears[j];
379 
380  const auto& other_ear = face_data.ears[other_index];
381 
382  const int64_t other_f = face_data.split_f[other_index];
383  const int64_t other_v =
384  m_spine_vids[other_index]; // v from the original spine that needs to be replaced
385  const int64_t se = split_spine_eids[j]; // new spine edge to replace
386 
387 
388  update_ids_in_ear(ear, f, f_old);
389 
390  auto fv = fv_accessor.index_access().vector_attribute(f);
391  auto fe = fe_accessor.index_access().vector_attribute(f);
392  auto ff = ff_accessor.index_access().vector_attribute(f);
393  fv = old_fv;
394  fe = old_fe;
395  ff = old_ff;
396  // correct old connectivity
397  for (size_t i = 0; i < 3; ++i) {
398  // if the original face edge was the other ear's edge then we replace it with thee
399  // spline
400  if (fe[i] == other_ear.eid) {
401  ff[i] = other_f;
402  fe[i] = split_edge_eid;
403  }
404 
405  // replace the input edge iwth the new edge for this triangle
406  if (fe[i] == m_operating_edge_id) {
407  fe[i] = se;
408  }
409 
410  // if i find the other vertex then i set it to be the new vertex
411  if (fv[i] == other_v) {
412  fv[i] = split_new_vid;
413  }
414  }
415  // assign each edge one face
416  ef_accessor.index_access().scalar_attribute(ear.eid) = f;
417  ef_accessor.index_access().scalar_attribute(se) = f;
418  // assign each vertex one face
419  vf_accessor.index_access().scalar_attribute(m_spine_vids[j]) = f;
420  }
421 
422  vf_accessor.index_access().scalar_attribute(v_opp) = new_fids[0];
423  vf_accessor.index_access().scalar_attribute(split_new_vid) = new_fids[0];
424 
425  ef_accessor.index_access().scalar_attribute(split_edge_eid) = new_fids[0];
426  vf_accessor.index_access().scalar_attribute(split_new_vid) = new_fids[0];
427 
428  // face neighbors on the other side of the spine are updated separately
429 
430  return;
431 }
