338 bool any_tag_present(const CellTag& tag1, const CellTag& tag2)
339 {
340 if (tag2.empty()) {

343 if (tag1.empty()) { // tag1 is ambient, tag2 is not

358 void sort_edges_by_length(std::vector<simplex::Edge>& edges)
359 {
360 std::sort(
361 edges.begin(),
362 edges.end(),
363 [this](const simplex::Edge& e1, const simplex::Edge& e2) {
364 double len1 = (m_vertex_attribute[e1.vertices()[0]].m_posf -
365 m_vertex_attribute[e1.vertices()[1]].m_posf)
366 .norm();
367 double len2 = (m_vertex_attribute[e2.vertices()[0]].m_posf -
368 m_vertex_attribute[e2.vertices()[1]].m_posf)
369 .norm();
