106  SplitRibBasicStrategy optype)
107 {
108  using VT = NewAttributeStrategy::VecType<T>;
109  switch (optype) {
110  default: [[fallthrough]];
111  case SplitRibBasicStrategy::Default:
112  if constexpr (std::is_same_v<T, double> || std::is_same_v<T, Rational>) {
113  return set_standard_split_rib_strategy(SplitRibBasicStrategy::Mean);
114  } else {
115  return set_standard_split_rib_strategy(SplitRibBasicStrategy::CopyTuple);
116  }
117  case SplitRibBasicStrategy::CopyTuple:
118  set_split_rib_strategy([&](const VT& a, const VT& b, const std::bitset<2>& bs) -> VT {
119  // if both are boundary then return a (failed link anyway but oh well)
120  // if a is boundary but b is interior get b though
121  if (!bs[1] && bs[0]) {
122  return b;
123  } else {
124  return a;
125  }
126  });
127  break;
128  case SplitRibBasicStrategy::CopyOther:
129  set_split_rib_strategy([&](const VT& a, const VT& b, const std::bitset<2>& bs) -> VT {
130  if (!bs[0] && bs[1]) {
131  return a;
132  } else {
133  return b;
134  }
135  });
136  break;
137  case SplitRibBasicStrategy::Mean:
138  set_split_rib_strategy([&](const VT& a, const VT& b, const std::bitset<2>& bs) -> VT {
139  if (bs[0] == bs[1]) {
140  return (a + b) / T(2);
141  } else if (bs[0]) {
142  return a;
143 
144  } else {
145  return b;
146  }
147  });
148  break;
149  case SplitRibBasicStrategy::Throw:
