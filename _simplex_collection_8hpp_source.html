146template <int N>
147inline SimplexCollection SimplexCollection::faces_from_simplex(const Simplex<N>& simplex)
148{
149 const auto& v = simplex.vertices();
150
151 SimplexCollection sc;
152
153 if constexpr (N == 1) {
154 // do nothing
155 } else if constexpr (N == 2) {
156 sc.m_v.emplace_back(Vertex(v[0]));
157 sc.m_v.emplace_back(Vertex(v[1]));
158 } else if constexpr (N == 3) {
159 sc.m_v.emplace_back(Vertex(v[0]));
160 sc.m_v.emplace_back(Vertex(v[1]));
161 sc.m_v.emplace_back(Vertex(v[2]));
162 sc.m_e.emplace_back(Edge(v[0], v[1]));
163 sc.m_e.emplace_back(Edge(v[0], v[2]));
164 sc.m_e.emplace_back(Edge(v[1], v[2]));
165 } else {
166 static_assert(N == 4);
167
168 sc.m_v.emplace_back(Vertex(v[0]));
169 sc.m_v.emplace_back(Vertex(v[1]));
170 sc.m_v.emplace_back(Vertex(v[2]));
171 sc.m_v.emplace_back(Vertex(v[3]));
172 sc.m_e.emplace_back(Edge(v[0], v[1]));
173 sc.m_e.emplace_back(Edge(v[0], v[2]));
174 sc.m_e.emplace_back(Edge(v[0], v[3]));
175 sc.m_e.emplace_back(Edge(v[1], v[2]));
176 sc.m_e.emplace_back(Edge(v[1], v[3]));
177 sc.m_e.emplace_back(Edge(v[2], v[3]));
178 sc.m_f.emplace_back(Face(v[0], v[1], v[2]));
179 sc.m_f.emplace_back(Face(v[0], v[1], v[3]));
180 sc.m_f.emplace_back(Face(v[0], v[2], v[3]));
181 sc.m_f.emplace_back(Face(v[1], v[2], v[3]));
182 }
183
184 return sc;
185}
