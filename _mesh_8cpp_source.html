24 std::vector<Tuple> Mesh::get_all(PrimitiveType type, const bool include_deleted) const
25 {
26  std::vector<Tuple> ret;
27 
28  if (static_cast<int8_t>(type) > top_cell_dimension()) return ret;
29 
30  const int64_t cap = capacity(type);
31 
32  const attribute::Accessor<char> flag_accessor = get_flag_accessor(type);
33  const attribute::CachingAccessor<char>& flag_accessor_indices = flag_accessor.index_access();
34  ret.reserve(cap);
35  for (size_t index = 0; index < cap; ++index) {
36  if (flag_accessor_indices.const_scalar_attribute(index) & 1)
37  ret.emplace_back(tuple_from_id(type, index));
38  else if (include_deleted)
39  ret.emplace_back();
40  }
41  return ret;
42 }
