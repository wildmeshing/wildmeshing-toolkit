278 auto run_single_queue = [&](Queue& Q, int task_id) {
279 Elem ele_in_queue;
280 while ([&]() { return Q.try_pop(ele_in_queue); }()) {
281 auto& [weight, op, tup, retry] = ele_in_queue;
282 if (!tup.is_valid(m)) {
283 continue;
284 }
285
286 std::vector<Elem> renewed_elements;
287 {
288 auto locked_vid = lock_vertices(
289 m,
290 tup,
291 task_id); // Note that returning `Tuples` would be invalid.
292 if (!locked_vid) {
293 retry++;
294 if (retry < max_retry_limit) {
295 Q.emplace(ele_in_queue);
296 } else {
297 retry = 0;
298 final_queue.emplace(ele_in_queue);
299 }
300 continue;
301 }
302 if (tup.is_valid(m)) {
303 if (!is_weight_up_to_date(
304 m,
305 std::tuple<double, Op, Tuple>(weight, op, tup))) {
306 operation_cleanup(m);
307 continue;
308 } // this can encode, in qslim, recompute(energy) == weight.
309 auto newtup = edit_operation_maps[op](m, tup);
310 std::vector<std::pair<Op, Tuple>> renewed_tuples;
311 if (newtup) {
312 renewed_tuples = renew_neighbor_tuples(m, op, newtup.value());
313 cnt_success++;
314 cnt_update++;
315 } else {
316 on_fail(m, op, tup);
317 cnt_fail++;
318 }
319 for (const auto& [o, e] : renewed_tuples) {
320 auto val = priority(m, o, e);
321 if (should_renew(val)) {
322 renewed_elements.emplace_back(val, o, e, 0);
323 }
324 }
325 }
326 operation_cleanup(m); // Maybe use RAII
327 }
328 for (auto& e : renewed_elements) {
329 Q.emplace(e);
330 }
331
332 if (stop.load(std::memory_order_acquire)) {
333 return;
334 }
335 if (cnt_success > stopping_criterion_checking_frequency) {
336 if (stopping_criterion(m)) {
337 stop.store(true);
338 return;
339 }
340 cnt_update.store(0, std::memory_order_release);
341 }
342 }
343 };
344
345 if (policy == ExecutionPolicy::kSeq) {
346 for (const auto& [op, e] : operation_tuples) {
347 if (!e.is_valid(m)) {
348 continue;
349 }
350 final_queue.emplace(priority(m, op, e), op, e, 0);
351 }
352 run_single_queue(final_queue, 0);
353 } else {
354 for (const auto& [op, e] : operation_tuples) {
355 if (!e.is_valid(m)) {
356 continue;
357 }
358 queues[get_partition_id(m, e)].emplace(priority(m, op, e), op, e, 0);
359 }
360 // Comment out parallel: work on serial first.
361 wmtk::threading::task_group tg;
362 for (int task_id = 0; task_id < queues.size(); task_id++) {
363 tg.run([&run_single_queue, &queues, task_id] {
364 run_single_queue(queues[task_id], task_id);
365 });
366 }
367 tg.wait();
368 logger().debug("Parallel Complete, remains element {}", final_queue.size());
369 run_single_queue(final_queue, 0);
370 }
371
372 logger().info(
373 "executed: {} | success / fail: {} / {}",
374 (int)cnt_success + (int)cnt_fail,
375 (int)cnt_success,
376 (int)cnt_fail);
377 return true;
378 }
