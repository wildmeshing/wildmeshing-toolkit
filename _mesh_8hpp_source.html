416 Tuple Mesh::switch_tuples(const Tuple& tuple, const ContainerType& sequence) const
417 {
418  static_assert(std::is_same_v<typename ContainerType::value_type, PrimitiveType>);
419  Tuple r = tuple;
420  const PrimitiveType top_type = top_simplex_type();
421  for (const PrimitiveType primitive : sequence) {
422  // for top level simplices we cannot navigate across boundaries
423  if (primitive == top_type && is_boundary(r)) {
424  assert(!is_boundary(r));
425  r = {};
426  return r;
427  }
428  r = switch_tuple(r, primitive);
429  }
430  return r;
431 }
