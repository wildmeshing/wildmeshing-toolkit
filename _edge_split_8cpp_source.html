124 std::pair<Tuple, Tuple> EdgeSplit::new_spine_edges(const Mesh& mesh, const Tuple& new_vertex)
125 {
126  // new_vertex is a spine edge on a face pointing to the new vertex, so we
127  // * PE -> new edge
128  // * PF -> other face
129  // * PE -> other spine edge

131  constexpr static PrimitiveType PF = PrimitiveType::Face;

133 
134  std::pair<Tuple, Tuple> ret;
135 
136  switch (mesh.top_simplex_type()) {

139  break;
140  }
141  case PF: {
142  ret = {new_vertex, mesh.switch_tuples(new_vertex, {PE, PF, PE})};
143  break;
144  }
145  case PT: {
146  ret = {new_vertex, mesh.switch_tuples(new_vertex, {PE, PF, PT, PF, PE})};
147  break;
148  }
149  case PrimitiveType::Vertex:
150  case PrimitiveType::HalfEdge:
151  default: throw std::runtime_error("Invalid top simplex");
152  }
